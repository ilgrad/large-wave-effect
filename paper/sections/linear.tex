\section{Toeplitz and circulant matrices: a unified language}\label{sec:setting}

The bead chain is a spring--mass lattice whose stiffness operator is shift invariant---hence
Toeplitz---and, under periodic boundary conditions, circulant (Figure~\ref{fig:chains}). We collect
the structures and the classical algorithms that organize the entire analysis.

\subsection{Toeplitz, Hankel, circulant}
A matrix is \emph{Toeplitz} if it is constant along diagonals, $A_{ij}=a_{i-j}$ ($2n-1$ scalars,
$O(n)$ storage); \emph{Hankel} if constant along anti-diagonals, $H_{ij}=h_{i+j}$; and
\emph{circulant} if Toeplitz with a periodic wrap, $C_{ij}=c_{(i-j)\bmod n}$. For $n=4$,
\[
T=\begin{pmatrix}a_0&a_{-1}&a_{-2}&a_{-3}\\ a_1&a_0&a_{-1}&a_{-2}\\
a_2&a_1&a_0&a_{-1}\\ a_3&a_2&a_1&a_0\end{pmatrix},\quad
H=\begin{pmatrix}h_0&h_1&h_2&h_3\\ h_1&h_2&h_3&h_4\\ h_2&h_3&h_4&h_5\\ h_3&h_4&h_5&h_6\end{pmatrix},\quad
C=\begin{pmatrix}c_0&c_3&c_2&c_1\\ c_1&c_0&c_3&c_2\\ c_2&c_1&c_0&c_3\\ c_3&c_2&c_1&c_0\end{pmatrix}.
\]
Reversing rows turns Toeplitz into Hankel, $H=JT$ with $J$ the exchange matrix; a Toeplitz matrix is
the finite compression of a convolution (Laurent) operator. The ring Laplacian
$\LN=\circm(2,-1,0,\dots,0,-1)$ is the periodic case of the segment Toeplitz
$T_N=\operatorname{tridiag}(-1,2,-1)$.

\subsection{The symbol and circulant diagonalization}
A banded Toeplitz/circulant matrix with entries $a_k$ has \emph{symbol}
$f(\theta)=\sum_k a_k\eu^{\mathrm{i}k\theta}$; for both $\LN$ and $T_N$,
\[ f(\theta)=2-2\cos\theta=4\sin^2(\theta/2). \]
Every circulant is a polynomial $C=\sum_k c_k P^k$ in the cyclic shift $P$, and the Fourier modes
$\varphi^{(r)}_j=N^{-1/2}\eu^{2\pi\mathrm{i}rj/N}$ are common eigenvectors of $P$
($P\varphi^{(r)}=\eu^{2\pi\mathrm{i}r/N}\varphi^{(r)}$). Hence
\begin{equation}\label{eq:diag}
  \LN=F_N^\ast\Lambda F_N,\qquad
  \lambda_r=f\!\left(\tfrac{2\pi r}{N}\right)=4\sin^2\tfrac{\pi r}{N},\qquad
  \omega_r=\sqrt{\lambda_r}=2\sin\tfrac{\pi r}{N},
\end{equation}
$r=0,\dots,N-1$. The discrete dispersion relation $\omega(\theta)=2|\sin(\theta/2)|$ is literally the
square root of the symbol, sampled at $\theta=2\pi r/N$ on the ring and at $\theta=\pi k/(N{+}1)$ on
the segment ($\lambda^{\mathrm{Dir}}_k=4\sin^2\frac{\pi k}{2(N+1)}$). The twofold degeneracy
$\omega_r=\omega_{N-r}$ on the ring---absent on the segment---is the source of the arithmetic
difficulty in Section~\ref{sec:prime}.

\subsection{Classical Toeplitz algorithms}
Determined by $2n-1$ scalars, a Toeplitz system admits operations far below the dense $O(n^3)$
(Table~\ref{tab:toeplitz}); three enter the analysis.

\noindent\textbf{Levinson--Durbin.} For symmetric Toeplitz $R=(r_{|i-j|})$ the order-$i$ system
$R^{(i)}a^{(i)}=(r_1,\dots,r_i)^\top$ is built recursively through the \emph{reflection coefficients}
$k_i$ (PARCOR): with $E_0=r_0$,
\begin{equation}\label{eq:levinson}
  k_i=\frac{r_i-\sum_{j=1}^{i-1}a^{(i-1)}_j r_{i-j}}{E_{i-1}},\quad
  a^{(i)}_j=a^{(i-1)}_j-k_i\,a^{(i-1)}_{i-j},\quad a^{(i)}_i=k_i,\quad E_i=(1-k_i^2)E_{i-1}.
\end{equation}
Recursion~\eqref{eq:levinson} is due to Levinson~\cite{Levinson} and, in the AR (Yule--Walker) case,
Durbin~\cite{Durbin}; one has $|k_i|<1$ for all $i$ iff $R\succ0$, making $\{k_i\}$ the natural
stable-filter parametrization. The implementation used here goes back to the author's 2012
study~\cite{Gradina12}.

\noindent\textbf{Trench / Gohberg--Semencul.} The inverse of a Toeplitz matrix is \emph{not} Toeplitz
but is determined by its first and last columns $u,v$~\cite{Trench,GS} (two $O(n^2)$ recursions); writing $L_x,U_x$
for the lower/upper triangular Toeplitz matrices generated by a vector $x$,
\begin{equation}\label{eq:gs}
  T^{-1}=\tfrac1{u_0}\bigl(L_u U_{v}-L_{Sv}\,U_{Su}\bigr),
\end{equation}
a difference of products of triangular Toeplitz factors ($S$ the shift), with $\det T$ obtained en
route. On the ring this collapses to the pseudoinverse $\LN^{+}=F_N^\ast\Lambda^{+}F_N$ (the constant
mode, $\LN\mathbf1=0$, is mapped to $0$) and, more generally, $f(\LN)=F_N^\ast f(\Lambda)F_N$ in
$O(N\log N)$---the route by which the propagators below are evaluated.

\noindent\textbf{Wiener--Hopf.} On the semi-infinite chain the stiffness is a one-sided Toeplitz
operator; its inversion is the Wiener--Hopf factorization $f(\theta)=f_+(\theta)f_-(\theta)$ of the
symbol, which links the finite-$N$ ceilings to the operator-theoretic (Szeg\H{o}) asymptotics of large
truncated Toeplitz matrices~\cite{BS}, Section~\ref{sec:upper}.

\subsection{Three dynamics and the central functional}
The same lattice carries the wave equation $\ddot u+\LN u=0$ (propagator $\cos(t\sqrt{\LN})$), the
discrete Schr\"odinger equation $\mathrm i\dot z=\LN z$ (propagator $\eu^{-\mathrm i t\LN}$, the phase
governed by $\lambda_r$ rather than $\omega_r=\sqrt{\lambda_r}$), and the dissipative heat flow
$\dot v=-\LN v$ (no large wave---the semigroup smooths). For the wave equation with the unit zero-mean
velocity impulse $u(0)=0$, $\dot u(0)=e_0-\tfrac1N\mathbf 1$, the response is the velocity Green's
function
\begin{equation}\label{eq:green}
  u_j(t)=G_N(j,t)=\frac1N\sum_{r=1}^{N-1}\frac{\sin(\omega_r t)}{\omega_r}\,\eu^{2\pi\mathrm{i}rj/N}.
\end{equation}
As $u$ is real and $\omega_r=\omega_{N-r}$, \eqref{eq:green} is a real almost-periodic function of $t$
with frequency set $\{\omega_r\}$, and the large wave is the operator-norm growth
$A_N=\sup_t\max_j|G_N(j,t)|$.

\section{Bohr's supremum and the ceiling \texorpdfstring{$U_N$}{U\_N}}

\begin{proposition}[Bohr ceiling]\label{prop:bohr}
For every $N$ and every node $j$,
\(
  \sup_t|G_N(j,t)|\le U_N,\ U_N:=\tfrac1{2N}\sum_{r=1}^{N-1}\csc(\pi r/N),
\)
with equality at $j=0$ provided the frequencies $\{\omega_r:1\le r\le\lfloor N/2\rfloor\}$ are
linearly independent over $\Q$.
\end{proposition}

\begin{proof}
Grouping the degenerate pair $r,\,N-r$ gives, at $j=0$,
$G_N(0,t)=\sum_{r=1}^{\lfloor N/2\rfloor}b_r\sin(\omega_r t)$ with $b_r=2/(N\omega_r)>0$
(the term $r=N/2$, if present, has coefficient $1/(N\omega_{N/2})$). The triangle inequality
gives $|G_N(j,t)|\le\frac1N\sum_{r=1}^{N-1}\omega_r^{-1}=U_N$ for all $j$. If the $\omega_r$ are
$\Q$-independent, then by Kronecker--Weyl the orbit $\{(\omega_r t)_r\}$ is dense in the torus, so
$\sin(\omega_r t)\to1$ simultaneously and the supremum at $j=0$ equals the bound. (Bohr's theorem:
$\sup_t\sum b_r\sin(\omega_r t)=\sum|b_r|$.)
\end{proof}

\section{The exact constant}

\begin{lemma}[Euler--Maclaurin asymptotics]\label{lem:asym}
$\displaystyle U_N=\frac1\pi\Big(\ln N+\gamma+\ln\frac2\pi\Big)+o(1)
   =\frac1\pi\ln N+0.039\,99\ldots+o(1).$
\end{lemma}

\noindent The leading term comes from the singularity $\csc(\pi r/N)\sim N/(\pi r)$ at small $r$,
which yields the harmonic sum $\sum 1/r\sim\ln$; the constant follows from the Euler--Maclaurin
correction. Numerically $U_N-\tfrac1\pi\ln N\to0.03999$ for $N\le 65536$
(\texttt{numerics/verify\_lemma3\_lebesgue.py}).

\section{Upper bounds: energy and the Szeg\H{o} symbol}\label{sec:upper}

A fixed-energy bound caps the response. With energy
$E=\tfrac12\dot u^\top\dot u+\tfrac12 u^\top\LN u$ and the exact trigonometric identity (the cotangent
sum, derived in Section~\ref{sec:zeta})
\begin{equation}\label{eq:csc2}
  \sum_{r=1}^{N-1}\frac1{\sin^2(\pi r/N)}=\frac{N^2-1}{3},
\end{equation}
Cauchy--Schwarz on the modal expansion gives $\|u(t)\|_\infty\le C\sqrt N\,\sqrt E$: at fixed energy
the deflection cannot exceed order $\sqrt N$, a ceiling attained by the time-reversed focusing
configuration (the variational maximum; \texttt{verify\_theorem1.py}). The same sum yields the exact
pseudo-inverse diagonal $(\LN^{+})_{jj}=\tfrac1N\sum_{r\ne0}\lambda_r^{-1}=\tfrac{N^2-1}{12N}$.

Asymptotically the eigenvalues follow the symbol through Szeg\H{o}'s theorem~\cite{GrenanderSzego,Gray}:
for continuous $F$,
\begin{equation}\label{eq:szego}
  \frac1N\sum_{r=0}^{N-1}F(\lambda_r)\xrightarrow{N\to\infty}\frac1{2\pi}\int_0^{2\pi}F\bigl(f(\theta)\bigr)\,d\theta,
  \qquad f(\theta)=4\sin^2(\theta/2),
\end{equation}
so spectral sums such as $U_N$ and $(\LN^{+})_{jj}$ are Riemann sums of the symbol; the logarithmic
growth of $U_N$ is exactly the borderline (logarithmically divergent) singularity
$f^{-1/2}\sim|\theta|^{-1}$ at $\theta=0$. (Szeg\H{o}'s theorem
proper requires continuous $F$; the \emph{singular} sums $U_N=\tfrac1{2N}\sum\lambda_r^{-1/2}$ and
$(\LN^{+})_{jj}=\tfrac1N\sum\lambda_r^{-1}$ are made rigorous by Euler--Maclaurin at the band edge,
Appendix~\ref{app:bernoulli}, the singularity producing the logarithm and the $N^2$ growth.) The propagators
$\cos(t\sqrt{\LN})$, $\eu^{-\mathrm i t\LN}$, $\LN^{-1/2}\sin(t\sqrt{\LN})$ are functions of the matrix
in the sense of Golub--Van Loan~\cite{GVL}, evaluated exactly by \eqref{eq:diag}; this puts the
operator-norm functionals $\|f_t(\LN)\|_{\ell^p\to\ell^q}$ on a rigorous footing.

\section{The spectral zeta function and the large-wave constants}\label{sec:zeta}

Define the spectral zeta of the ring Laplacian (zero mode removed),
\begin{equation}\label{eq:zeta}
  \zeta_{\LN}(s)=\sum_{r=1}^{N-1}\lambda_r^{-s}=4^{-s}\sum_{r=1}^{N-1}\csc^{2s}\tfrac{\pi r}{N},
  \qquad \Re s>0 .
\end{equation}
At integer $s$ this is a cosecant power sum with a closed form in $N$ (Bernoulli numbers,
Appendix~\ref{app:bernoulli}); the first two are
\[
  \zeta_{\LN}(1)=\frac{N^2-1}{12},\qquad \zeta_{\LN}(2)=\frac{(N^2-1)(N^2+11)}{720}.
\]
The first is elementary: from the classical identity $\sum_{r=1}^{N-1}\cot^2\frac{\pi r}{N}=\frac{(N-1)(N-2)}3$
and $\csc^2=1+\cot^2$,
\[
  \sum_{r=1}^{N-1}\csc^2\tfrac{\pi r}{N}=(N-1)+\frac{(N-1)(N-2)}3=\frac{N^2-1}3,
  \qquad\text{so}\qquad \zeta_{\LN}(1)=\tfrac14\sum_{r}\csc^2\tfrac{\pi r}{N}=\frac{N^2-1}{12},
\]
and $\zeta_{\LN}(2)$ follows the same way from $\sum_r\csc^4=\frac{(N^2-1)(N^2+11)}{45}$ (all integer values
are even polynomials in $N$, \texttt{verify\_spectral\_zeta.py}).
The two large-wave constants are \emph{special values of this single object}:
\begin{equation}\label{eq:zeta-values}
  U_N=\frac{\zeta_{\LN}(1/2)}{N},\qquad M_N^2=\frac{2E_0\,\zeta_{\LN}(1)}{N},
\end{equation}
the first because $\csc(\pi r/N)=2\lambda_r^{-1/2}$, the second because $(\LN^{+})_{jj}=\zeta_{\LN}(1)/N$.
Thus the Bohr ceiling (the large wave, $A_N\sim U_N$) and the variational ceiling $M_N$
(Section~\ref{sec:upper}) are the values of $\zeta_{\LN}$ at $s=\tfrac12$ and $s=1$.

The two growth laws are then dictated by the analytic structure of $\zeta_{\LN}$. By Szeg\H{o}'s
theorem the normalized zeta converges to the symbol integral,
\begin{equation}\label{eq:zeta-density}
  \frac1N\,\zeta_{\LN}(s)\xrightarrow{N\to\infty}
  I(s)=\frac1{2\pi}\int_0^{2\pi}(2-2\cos\theta)^{-s}\,d\theta
  =\frac{4^{-s}\,\Gamma(\tfrac12-s)}{\sqrt{\pi}\,\Gamma(1-s)},\qquad \Re s<\tfrac12,
\end{equation}
which has a simple pole at $s=\tfrac12$ (Figure~\ref{fig:zeta}). The edge of convergence is exactly the
logarithmic singularity producing $U_N=\zeta_{\LN}(\tfrac12)/N=\tfrac1\pi\ln N+O(1)$ (Lemma~\ref{lem:asym}).
At $s=1$ the density diverges, and indeed $\zeta_{\LN}(1)\sim N^2/12$ yields the $\sqrt N$ variational
ceiling $M_N\sim\sqrt{N/6}$. \emph{The large wave sits precisely at the convergence edge $s=\tfrac12$ of
the limiting spectral density $I(s)$}; the finite-$N$ $\zeta_{\LN}$ is itself an entire finite sum, and
it is the $N\to\infty$ density that carries the pole. The finite-$N$ zeta is an Epstein/Hurwitz-type
object whose integer values are polynomials in $N^2$ and whose half-integer value carries the logarithm.
All identities are verified (\texttt{verify\_spectral\_zeta.py}). At the opposite end, the spectral
determinant is the $s=0$ \emph{derivative}: $\zeta_{\LN}'(0)=-\log\det{}'\LN$ with
$\det{}'\LN=\prod_{r\ne0}\lambda_r=N^2$ (from $\prod_{r=1}^{N-1}2\sin(\pi r/N)=N$), so by Kirchhoff's
matrix-tree theorem the cycle $C_N$ has exactly $\det{}'\LN/N=N$ spanning trees---a graph invariant read
off $\zeta_{\LN}'(0)$ (\texttt{verify\_spanning\_trees.py}).

\section{Operator-norm hierarchy and the variational principle}\label{sec:opnorm}

The large wave is one corner of a hierarchy of operator norms of the velocity propagator
$S_N(t)=f_t(\LN)$ on the zero-mean subspace $\mathcal H_0=\{x\in\R^N:\sum_jx_j=0\}$, where
$f_t(\lambda)=\sin(t\sqrt\lambda)/\sqrt\lambda$ for $\lambda>0$ and $f_t(0)=0$ (circulant on
$\mathcal H_0$, kernel $G_N$; the constant mode is excluded since $\LN\mathbf1=0$). Its time-supremum in
different $\ell^p\to\ell^q$ norms obeys \emph{three distinct growth laws} (Table~\ref{tab:opnorm}, all
verified by \texttt{verify\_operator\_norms.py}):
\begin{equation}\label{eq:opnorm}
\begin{aligned}
  \sup_t\|S_N(t)\|_{2\to2}&=\frac1{2\sin(\pi/N)}\sim\frac{N}{2\pi},\\
  \sup_t\|S_N(t)\|_{2\to\infty}&\le\sqrt{\tfrac{\zeta_{\LN}(1)}{N}}\sim\sqrt{\tfrac N{12}},\\
  \sup_t\|S_N(t)\|_{1\to\infty}&=A_N\le U_N\sim\tfrac1\pi\ln N .
\end{aligned}
\end{equation}
\emph{Derivations.} $S_N(t)$ is symmetric with eigenvalues $f_t(\lambda_r)=\sin(t\omega_r)/\omega_r$, so
its $\ell^2\!\to\!\ell^2$ norm is $\max_r|\sin(t\omega_r)|/\omega_r$, whose supremum over $t$ is
$1/\omega_1=1/(2\sin\tfrac\pi N)$---the slowest mode reaching $\sin(t\omega_1)=\pm1$; this is exact and
unconditional. For $\ell^2\!\to\!\ell^\infty$, by translation invariance the norm is the $\ell^2$-length
of one row of $G_N$, and Parseval gives
\[
  \|G_N(\cdot,t)\|_2^2=\frac1N\sum_{r=1}^{N-1}\frac{\sin^2(t\omega_r)}{\omega_r^2}
  \ \le\ \frac1N\sum_{r}\omega_r^{-2}=\frac{\zeta_{\LN}(1)}N,
\]
hence $\sup_t\|S_N(t)\|_{2\to\infty}\le\sqrt{\zeta_{\LN}(1)/N}\sim\sqrt{N/12}$, with equality iff all
$\sin^2(t\omega_r)=1$ simultaneously---rational independence ($N$ prime, $2^m$). Finally
$\|S_N(t)\|_{1\to\infty}=\max_{j,k}|S_N(t)_{jk}|=\max_m|G_N(m,t)|$, whose time-supremum is, by definition,
$A_N\le\sum_r|a_r|=U_N$. The three corners thus read the spectral sums $\sum_r\omega_r^{-p}$ ($p=1,2$)
through different lenses; the $\ell^1\!\to\!\ell^\infty$ corner---weakest input, strongest output---is
precisely the large wave $A_N$.

\emph{Variational principle.} The energy norm has the direct reading: maximizing the deflection at a
node over the energy sphere $E(u_0,v_0)\le E_0$ on $\mathcal H_0$ is a constrained quadratic problem solved by Lagrange
multipliers,
\[
  \sup_{E(u_0,v_0)\le E_0}\ \max_j|u_j(t)| = \sqrt{2E_0\,(\LN^{+})_{jj}}=\sqrt{\tfrac{E_0(N^2-1)}{6N}}=M_N,
\]
\emph{independent of $t$} (the focusing configuration is the time-reversed peak). This is the
operator-norm/variational face of the energy ceiling, and via \eqref{eq:zeta-values} both $M_N$ and
$U_N$ are special values of the spectral zeta. The Szeg\H{o} symbol $f(\theta)=4\sin^2(\theta/2)$
controls all three norms through the spectral sums $\sum_r\omega_r^{-p}$.

\section{Independence theorems and unconditional growth}\label{sec:prime}

\begin{theorem}[odd prime]\label{thm:prime}
Let $p$ be an odd prime. Then $\{\sin(\pi r/p):1\le r\le\tfrac{p-1}2\}$ is linearly independent
over $\Q$; hence $A_p=U_p\sim\tfrac1\pi\ln p$.
\end{theorem}

\begin{proof}
Put $\eta=\zeta_{2p}=\eu^{\mathrm{i}\pi/p}$. As $p$ is odd, $[\Q(\eta):\Q]=\varphi(2p)=p-1$, so
$\{1,\eta,\dots,\eta^{p-2}\}$ is a $\Q$-basis, and $\eta^{p}=\eu^{\mathrm{i}\pi}=-1$. From
$\eta^{-r}=\eta^{2p-r}=\eta^{p}\eta^{p-r}=-\eta^{p-r}$ we get $2\mathrm{i}\sin(\pi r/p)=\eta^r-\eta^{-r}=\eta^r+\eta^{p-r}$.
Suppose $\sum_{r=1}^{(p-1)/2}c_r\sin(\pi r/p)=0$ with $c_r\in\Q$. Then
$\sum_r c_r\eta^r+\sum_r c_r\eta^{p-r}=0$. As $r$ runs over $\{1,\dots,\tfrac{p-1}2\}$ the exponents
$r$ fill $\{1,\dots,\tfrac{p-1}2\}$ and $p-r$ fill $\{\tfrac{p+1}2,\dots,p-1\}$, together covering
$\{1,\dots,p-1\}$ bijectively, so $\sum_{m=1}^{p-1}d_m\eta^m=0$ with $d_m=c_m$ or $c_{p-m}$.
Finally $\Phi_{2p}(\eta)=\sum_{j=0}^{p-1}(-1)^j\eta^j=0$ gives $\eta^{p-1}=-\sum_{j=0}^{p-2}(-1)^j\eta^j$,
whose constant term is $-1\ne0$; equating coordinates in $\{1,\eta,\dots,\eta^{p-2}\}$ forces
$d_{p-1}=0$ and then all $d_m=0$, i.e. all $c_r=0$. The conclusion follows from
Proposition~\ref{prop:bohr}. \end{proof}

\begin{theorem}[powers of two]\label{thm:two}
For $N=2^m$, $\{\sin(\pi r/N):1\le r\le N/2\}$ is $\Q$-independent; hence $A_N=U_N\sim\tfrac1\pi\ln N$.
\end{theorem}

\begin{proof}
Now $\eta=\zeta_{2N}$, $\Phi_{2N}(x)=x^{N}+1$ (degree $\varphi(2N)=N$), so $\{1,\dots,\eta^{N-1}\}$ is a
basis and $\eta^{N}=-1$. Again $2\mathrm{i}\sin(\pi r/N)=\eta^r+\eta^{N-r}$, and for $1\le r\le N/2$ the
exponents $r$ and $N-r$ lie in $\{1,\dots,N-1\}$ \emph{without reduction}; they are distinct basis
powers (the index $N/2$ occurring with coefficient $2c_{N/2}$). Linear independence is immediate.
\end{proof}

An exact integer-rank certificate validates the prime case (\texttt{verify\_lemma1\_prime.py}); the
general rank formula is checked separately by \texttt{verify\_independence\_rank.py}.

\medskip\noindent\textbf{Worked example (prime $N=5$).} The two distinct frequencies
$\omega_1=2\sin36^\circ=1.1756$ and $\omega_2=2\sin72^\circ=1.9021$ stand in the ratio
$\omega_2/\omega_1=2\cos36^\circ=\varphi=\tfrac{1+\sqrt5}2$, the \emph{golden ratio}---irrational, so the
relation lattice is trivial ($\Lambda_5=\{0\}$) and Theorem~\ref{thm:prime} applies. The node-$0$ response
is the two-tone signal $G_5(0,t)=\tfrac25\bigl(\omega_1^{-1}\sin\omega_1 t+\omega_2^{-1}\sin\omega_2 t\bigr)$,
and by Kronecker both sines reach $1$ at a common near-alignment time, so
\[
  A_5=U_5=\tfrac15\bigl(\csc36^\circ+\csc72^\circ\bigr)=0.5506,
\]
the supremum equal to the ceiling, approached arbitrarily closely (in contrast to the composite $N=12$ below;
\texttt{verify\_exact\_AN.py}).

The low modes carry most of the ceiling, and---crucially---a \emph{long} independent prefix can always be
aligned. Let $M(N)$ be the largest $M$ with $\{\omega_1,\dots,\omega_M\}$ $\Q$-independent.

\begin{lemma}[prefix independence]\label{lem:prefix}
$M(N)=\tfrac12\varphi(2N)$: the first $\tfrac12\varphi(2N)$ frequencies
$\{\omega_r=2\sin(\pi r/N):1\le r\le\tfrac12\varphi(2N)\}$ are $\Q$-independent.
\end{lemma}

\begin{proof}
A nontrivial relation $\sum_{r=1}^K k_r\sin(\pi r/N)=0$ with $k_K\ne0$ ($K$ its largest index)---a
vanishing sum of roots of unity~\cite{CJ}---gives, with $\zeta=\zeta_{2N}$ and $P(x)=\sum_{r=1}^K k_r x^r$,
\[
  Q(x)=x^K\bigl(P(x)-P(1/x)\bigr)=\sum_{r=1}^K k_r\bigl(x^{K+r}-x^{K-r}\bigr)\in\Z[x],
\]
nonzero of degree $2K$ (leading coefficient $k_K$), with $Q(\zeta)=\zeta^K\sum_r k_r(\zeta^r-\zeta^{-r})=0$.
Hence $\Phi_{2N}\mid Q$ and $\varphi(2N)=\deg\Phi_{2N}\le2K$. But $Q$ is \emph{anti}-palindromic,
$x^{2K}Q(1/x)=-Q(x)$, whereas $\Phi_{2N}$ is palindromic; so $2K=\varphi(2N)$ is impossible (it would force
$Q=k_K\Phi_{2N}$, simultaneously palindromic and anti-palindromic, hence $0$). Thus $2K>\varphi(2N)$, and as
both are even, $K\ge\tfrac12\varphi(2N)+1$. No relation has all its indices $\le\tfrac12\varphi(2N)$, so
$\{\omega_1,\dots,\omega_{\varphi(2N)/2}\}$ are independent and $M(N)\ge\tfrac12\varphi(2N)$; with
$M(N)\le\operatorname{rank}_\Q\{\omega_r\}=\tfrac12\varphi(2N)$ (Theorem~\ref{thm:qrank}), equality holds. \end{proof}

\begin{theorem}[order]\label{thm:order}
$A_N\sim\tfrac1\pi\ln N$ for \emph{every} $N$; precisely $A_N=\tfrac1\pi\ln N+O(\ln\ln\ln N)$.
\end{theorem}

\begin{proof}
The upper bound is $A_N\le U_N=\tfrac1\pi\ln N+O(1)$ (Proposition~\ref{prop:bohr}, Lemma~\ref{lem:asym}). For
the lower bound, write $b_r=2/(N\omega_r)>0$, $L_{\mathrm{pre}}=\sum_{r=1}^M b_r$, $M=\tfrac12\varphi(2N)$.
By Lemma~\ref{lem:prefix} the $\omega_1,\dots,\omega_M$ are $\Q$-independent, so by Kronecker there are
times $t$ with $\sin(\omega_r t)\to1$ for all $r\le M$ at once; at such $t$, since the remaining
$\sin(\omega_r t)\ge-1$,
\[
  A_N\ \ge\ \sup_t G_N(0,t)\ \ge\ L_{\mathrm{pre}}-\!\!\sum_{r=M+1}^{\lfloor N/2\rfloor}\!\!b_r\ =\ 2L_{\mathrm{pre}}-U_N .
\]
As $\sin(\pi r/N)\le\pi r/N$, $\,b_r=1/(N\sin(\pi r/N))\ge1/(\pi r)$, so
$L_{\mathrm{pre}}\ge\tfrac1\pi\sum_{r=1}^M\tfrac1r\ge\tfrac1\pi\ln M$, giving
\[
  A_N\ \ge\ \tfrac2\pi\ln M-U_N\ =\ \tfrac1\pi\ln\frac{M^2}{N}-O(1)
        \ =\ \tfrac1\pi\Bigl(2\ln\tfrac{\varphi(2N)}2-\ln N\Bigr)-O(1).
\]
By the Rosser--Schoenfeld bound $\varphi(2N)\ge 2N/\!\left(\eu^{\gamma}\ln\ln 2N+3/\ln\ln 2N\right)$~\cite{RS62},
the bracket is $\ln N-O(\ln\ln\ln N)$; hence $A_N\ge\tfrac1\pi\ln N-O(\ln\ln\ln N)$, and with the upper
bound $A_N=\tfrac1\pi\ln N+O(\ln\ln\ln N)$ (\texttt{verify\_order\_theorem.py}). \end{proof}

\noindent The former obstruction---the destructive interference of the out-of-prefix modes---is dispatched
by the crude bound $\sin\ge-1$, which suffices because Lemma~\ref{lem:prefix} makes the alignable prefix
carry $2L_{\mathrm{pre}}-U_N=(1-o(1))U_N$ of the ceiling. The exact saturation locus $A_N=U_N$ is the
sharper question settled by Theorem~\ref{thm:classification}.

\begin{remark}[the defect: an odd-part scaling law]\label{rem:defect}
The same estimate controls the deficit on the non-saturating $N$: $U_N-A_N\le2(U_N-L_{\mathrm{pre}})=
2\sum_{r=\frac12\varphi(2N)+1}^{\lfloor N/2\rfloor}b_r=\tfrac2\pi\ln\tfrac{N}{\varphi(2N)}+O(1)$ (the tail of the
ceiling beyond the prefix; $b_{N/2}=\tfrac1{2N}$ when $N$ is even).
Writing $N=2^a m$ with $m$ odd, $N/\varphi(2N)=m/\varphi(m)$ is \emph{independent of $a$}, so the bound is
the same throughout each family $2^a m$:
\[
  U_N-A_N\ \le\ \tfrac2\pi\ln\tfrac{m}{\varphi(m)}+O(1),
\]
and the exact defect converges within the family to a limit depending only on the odd part $m$ (numerically
$U_N-A_N<0.18$ for composite $N\le60$, e.g.\ $0.05,0.08,0.09,0.10$ along $m=3$; \texttt{verify\_exact\_AN.py}).
Globally, since $m/\varphi(m)\sim\eu^{\gamma}\ln\ln m$ along primorials, this is $O(\ln\ln\ln N)$---slower than
any iterate of the logarithm. On all the numerical evidence the defect is \emph{not} bounded, and for a
clean reason: the \emph{excess} $A_N-L_{\mathrm{pre}}$ stays in $[0.05,0.12]$ across every tested $N$
($\omega(m)\le4$, up to $N=1155$ with prefix dimension $240$; the optimized $A_N$ is a lower bound, so
these are lower estimates of the excess), so numerically
\[
  U_N-A_N=(U_N-L_{\mathrm{pre}})-(A_N-L_{\mathrm{pre}})=\tfrac1\pi\ln\tfrac{m}{\varphi(m)}-O(1)\longrightarrow\infty
\]
along primorials ($\Theta(\ln\ln\ln N)$, \texttt{verify\_defect\_growth.py}), making the proven
$O(\ln\ln\ln N)$ upper bound sharp in order whichever alternative of the excess dichotomy holds (the deficit
grows for any excess that is $o(U_N-L_{\mathrm{pre}})$, indeed for any $A_N-L_{\mathrm{pre}}=c\,(U_N-L_{\mathrm{pre}})$
with $c<1$). A lower bound of the \emph{sharp} form $U_N-A_N\ge\tfrac1\pi\ln(m/\varphi(m))-O(1)$ is equivalent
to the \emph{excess lemma} $A_N\le L_{\mathrm{pre}}+O(1)$ (alternative \textup{(O1)} below)---the long alignable
prefix captures $A_N$ up to a constant. By the Kronecker--Weyl identification \eqref{eq:subtorus}, $A_N=\max_{\theta\in\mathbb T_N}\sum_r
b_r\sin\theta_r$ over the subtorus cut out by the relation lattice, so the lemma is precisely a
\emph{quantitative comparison on the relation-constrained subtorus}: the constrained maximum exceeds the
free-prefix maximum $L_{\mathrm{pre}}$ by only $O(1)$. This is a sup-side cancellation statement---a primal
analogue of the McGehee--Pigno--Smith/Konyagin $L^1$ lower bound~\cite{MPS81,Konyagin81}, and in the same
circle as the Chowla cosine problem on extrema of cosine sums~\cite{Bedert25} (indeed
$A_N=\tfrac2N\sup_t\int_0^t C(s)\,\mathrm ds$ is the maximal antiderivative of the cosine sum
$C(s)=\sum_r\cos(\omega_r s)$, the even-$N$ Nyquist mode at half weight)---but it runs in the opposite
direction (an \emph{upper} bound on a \emph{sup}, driven by the relation lattice rather than by the number
of frequencies). The recent spectral-to-cut resolution of that problem---Jin--Milojevi\'c--Tomon--Zhang~\cite{JMTZ25}
(and Bedert~\cite{Bedert25}), via Cayley-graph eigenvalues, with cut-norm/Grothendieck duality~\cite{AlonNaor06}
the matching tool---is the natural line of attack but does not transcribe: it \emph{lower}-bounds an
extremum's magnitude (small eigenvalue $\Rightarrow$ clique), whereas we need an \emph{upper} bound on a
constrained joint maximum; the frequencies $2\sin(\pi r/N)$ are not characters of a cyclic group; and the
dependent band's own supremum is its \emph{full} weight ($\max_\psi\sum_{r>M}b_r\sin\phi_r=\sum_{r>M}b_r$ for
$\omega(m)=2$; \texttt{verify\_excess\_omega2.py}), so the bound is forced by the prefix \emph{trade-off}, not
by any single sup. No off-the-shelf theorem supplies it, and the evidence is genuinely two-edged: the raw excess looks bounded,
yet the \emph{ratio} $(A_N-L_{\mathrm{pre}})/(U_N-L_{\mathrm{pre}})\approx0.31$ is stable across
$\omega(m)=2,3,4$, leaving open whether the excess is $O(1)$ or grows like $c\,(U_N-L_{\mathrm{pre}})$---the
dichotomy of Conjecture~\ref{conj:excess}.

The lemma's content is sharply localized. Since $A_N\le U_N$ and $A_N\ge2L_{\mathrm{pre}}-U_N$ (Kronecker),
one has $|A_N-L_{\mathrm{pre}}|\le U_N-L_{\mathrm{pre}}=\tfrac1\pi\ln\tfrac m{\varphi(m)}+O(1)$
unconditionally; so the excess lemma is \emph{trivial whenever the deficit is bounded}---i.e.\ whenever
$\omega(m)$ is bounded---and acquires content only as $\omega(m)\to\infty$. In particular it holds outright
when the odd part is a prime power ($\omega(m)\le1$): for $N=2p$ the single dependent mode $\omega_p=2$ has
weight $\tfrac1{2N}$, giving the sharp two-sided $|A_N-L_{\mathrm{pre}}|\le\tfrac1{2N}$
(Proposition~\ref{prop:2p}); for $N=4p$ the two dependent modes $\{2p-1,2p\}$ give
$|A_N-L_{\mathrm{pre}}|\le U_N-L_{\mathrm{pre}}=\tfrac1N\bigl(\sec\tfrac\pi{4p}+\tfrac12\bigr)=O(1/N)$, and
the same bookkeeping settles any family with $O(1)$ dependent modes (\texttt{verify\_excess\_smallcases.py}).
The genuinely open case is $\omega(m)\ge2$, where the dependent band carries $\Theta(N)$ modes of total
weight $\to\infty$ yet the excess stays $\approx0.1$; the accessible range ($\omega(m)\le3$) shows the small
excess is a \emph{balance} of a positive dependent net ($\approx0.17$) against a negative prefix deficit
($\approx-0.08$) rather than near-total sign cancellation, so the asymptotic mechanism remains conjectural
(Conjecture~\ref{conj:excess}).
For small $N$ the strict gap
$A_N<U_N$ is \emph{certified rigorously}: the Lipschitz-grid certificate gives
$U_N-A_N\ge0.049,0.057,0.030,0.082$ for $N=6,9,12,15$ (\texttt{verify\_defect\_certified.py}), and a
trigonometric Lasserre moment--SOS sharpens it (tight at $N=15$: $U_N-A_N=0.1475$, matching the optimizer
to five decimals, at solver tolerance) and reaches $N=105$
($M=24$), though too loosely there to certify the growth (\texttt{verify\_defect\_moment\_sos.py}). The exact
value is algebraic per $N$ (Proposition~\ref{prop:closed}) with no universal closed form.
\end{remark}

The numerical picture condenses into a single clean implication.

\begin{proposition}[conditional defect law]\label{prop:defect-cond}
Assume the two-sided excess lemma: there is an absolute constant $C$ with
$|A_N-L_{\mathrm{pre}}|\le C$ for all $N$. Then, writing $N=2^am$ with $m$ odd,
\[
  U_N-A_N\ =\ \tfrac1\pi\ln\frac{m}{\varphi(m)}\ +\ O(1):
\]
the deficit depends, to $O(1)$, only on the odd part of $N$, is unbounded, and is
$\Theta(\ln\ln\ln N)$ along primorials---so the proven $O(\ln\ln\ln N)$ upper bound is sharp in order.
\end{proposition}

\begin{proof}
$U_N-A_N=(U_N-L_{\mathrm{pre}})-(A_N-L_{\mathrm{pre}})$; the second bracket is $O(1)$ by hypothesis, and
the first equals $\tfrac1\pi\ln\bigl(N/\varphi(2N)\bigr)+O(1)=\tfrac1\pi\ln\bigl(m/\varphi(m)\bigr)+O(1)$
(the tail computation of Remark~\ref{rem:defect}); along primorials $m/\varphi(m)\sim\eu^\gamma\ln\ln m$.
\end{proof}

\noindent The hypothesis is exactly what the experiments support ($A_N-L_{\mathrm{pre}}\in[0.05,0.12]$
through $\omega(m)=4$, prefix dimension $240$); proving it is the open core.

\emph{A per-relation building block.} The elementary two-mode trade-off behind the excess admits a clean
closed bound.
\begin{lemma}[per-relation excess bound]\label{lem:triplet}
For all $\alpha,\beta,\gamma>0$,
\[
  g(\alpha,\beta,\gamma):=\max_{u,v\in\R}\bigl[\gamma\sin(u+v)-\alpha(1-\sin u)-\beta(1-\sin v)\bigr]
  \ \le\ \frac{\gamma^2}{2}\Bigl(\frac1\alpha+\frac1\beta\Bigr).
\]
Hence a single support-$3$ relation $\omega_c=\omega_a+\omega_b$ in isolation (so $\phi_c=\psi_a+\psi_b$ on
the subtorus) has excess exactly $g(b_a,b_b,b_c)$; since $b_x=1/(N\sin(\pi x/N))$ gives $1/b_a+1/b_b=1/b_c$,
this is at most $b_c/2$.
\end{lemma}
\begin{proof}
Put $p=1-\sin u,\ q=1-\sin v\in[0,2]$. Maximizing $\sin(u+v)=\sin u\cos v+\cos u\sin v$ over the signs of
$\cos u=\pm\sqrt{2p-p^2}$, $\cos v=\pm\sqrt{2q-q^2}$ gives
$\sin(u+v)\le|1-p|\sqrt{2q-q^2}+|1-q|\sqrt{2p-p^2}\le\sqrt{2p}+\sqrt{2q}$, using $|1-p|\le1$ on $[0,2]$ and
$\sqrt{2p-p^2}\le\sqrt{2p}$. Therefore
$g(\alpha,\beta,\gamma)\le\max_{p,q\ge0}\bigl[\gamma(\sqrt{2p}+\sqrt{2q})-\alpha p-\beta q\bigr]
=\frac{\gamma^2}{2\alpha}+\frac{\gamma^2}{2\beta}$ (separable, optimum $\sqrt{2p}=\gamma/\alpha$). Finally
$1/b_a+1/b_b=N\bigl(\sin\tfrac{\pi a}N+\sin\tfrac{\pi b}N\bigr)=N\sin\tfrac{\pi c}N=1/b_c$ (as
$\omega_c=\omega_a+\omega_b$), so $g(b_a,b_b,b_c)\le\tfrac{b_c^2}2\cdot\tfrac1{b_c}=\tfrac{b_c}2$
(\texttt{verify\_excess\_decomposition.py}).
\end{proof}

\begin{lemma}[the clean support-$3$ relations for $3\mid N$]\label{lem:cleancomb}
For $3\mid N$, the only solutions of $\omega_a+\omega_b=\omega_c$ with $1\le a<b<c\le N/2$ are
$a+b=N/3$, $c=N/3+a$. Hence the clean triplets $(a,\,N/3-a,\,N/3+a)$ are vertex-disjoint.
\end{lemma}
\begin{proof}
With $\zeta=\eu^{\mathrm i\pi/N}$ one has $\omega_r=-\mathrm i(\zeta^r-\zeta^{-r})$, so the relation is
$\operatorname{Im}(\zeta^a+\zeta^b-\zeta^c)=0$, i.e.\ (using $-\zeta^{-r}=\zeta^{N-r}$) the vanishing sum of
six $2N$-th roots of unity $\zeta^a+\zeta^{N-a}+\zeta^b+\zeta^{N-b}+\zeta^{N+c}+\zeta^{2N-c}=0$. By the
Conway--Jones classification of vanishing sums of roots of unity~\cite{CJ}, such a six-term sum is a union of
rotated minimal relations, and the only partitions of $6$ are $R_2^3$ (three antipodal pairs $\{x,x+N\}$) and
$R_3^2$ (two equilateral triples $\{x,x+\tfrac{2N}3,x+\tfrac{4N}3\}$). An antipodal pair within
$\{a,N-a,b,N-b,N+c,2N-c\}$ forces $c\in\{a,N-a\}$, impossible for distinct $a,b,c\in[1,N/2]$ (the case
$c=N-a$ needs $c=N/2$, collapsing $N+c=2N-c$); so the sum is $R_3^2$. The partner $a+\tfrac{2N}3$ of $a$ must
lie in $\{b,N-a,N-b,N+c,2N-c\}$; since $b-a<\tfrac N2<\tfrac{2N}3$, the cases $a+\tfrac{2N}3\in\{b,N+c,2N-c\}$
put the third triple member outside $[1,N/2]$ (or force $c=N/2$), and $a+\tfrac{2N}3=N-a$ needs $a=N/6=b$;
only $a+\tfrac{2N}3=N-b$ survives, giving $a+b=N/3$ and $c=N/3+a$. Conversely the identity
$\omega_{N/3+a}=\omega_a+\omega_{N/3-a}$ (i.e.\ $\sin(\tfrac\pi3+x)=\sin x+\sin(\tfrac\pi3-x)$,
$x=\pi a/N$; the collapse $2\sin\tfrac\pi6=1$) gives these relations. Vertex-disjointness is the involution
$a\mapsto N/3-a$ (verified exhaustively to $N=315$, \texttt{verify\_excess\_chain\_split.py}).
\end{proof}

\noindent This controls each clean relation \emph{in isolation}. For $3\mid N$ the \emph{coupled} triplets
separate too: by Lemma~\ref{lem:cleancomb} the clean (support-$3$) relations are exactly the vertex-disjoint
triples $(a,\,N/3-a,\,N/3+a)$, so partitioning the prefix penalty by any
$\lambda\in[0,1)$ and applying Lemma~\ref{lem:triplet} at weights $(1-\lambda)b_a,(1-\lambda)b_b$ with
$\sup(X{+}Y)\le\sup X+\sup Y$ yields the rigorous reduction
\[
  A_N-L_{\mathrm{pre}}\ \le\ \frac1{1-\lambda}\sum_{c\in T}\frac{b_c}2\ +\ F_{\mathrm{comb}}(\lambda),\qquad
  F_{\mathrm{comb}}(\lambda)=\max_{\psi}\Bigl[\sum_{r\in K}b_r\sin\phi_r-\lambda\!\sum_{s\le M}b_s(1-\sin\psi_s)\Bigr],
\]
splitting the dependent band into the disjoint triplets $T$ and the higher-support \emph{comb} generators
$K$. The triplet sum is bounded: for square-free $N=3p$, $c=\tfrac{2N}3-a$ makes $\sum_{c\in T}\tfrac{b_c}2$
a Riemann sum of $\tfrac12\int_{1/2}^{2/3}\csc(\pi x)\,dx=\tfrac{\ln3}{4\pi}\approx0.087$, and it stays
$<0.1$ across the tested $3\mid N$ (\texttt{verify\_excess\_chain\_split.py}). For square-free $N=3p$ the comb
block is a \emph{single} generator---the period-$3$ telescoping
$\sum_{r\le N/2,\,3\nmid r}(-1)^{\lfloor r/3\rfloor}\omega_r=0$ of $1+\zeta_3+\zeta_3^2=0$, defining the mode
$r=N/3$ of weight $b_{N/3}=2/(N\sqrt3)\to0$---so there
$A_N-L_{\mathrm{pre}}\le\tfrac{\ln3}{4\pi}+o(1)$ \emph{unconditionally}, with no signed cancellation needed.
In general the $3\mid N$ excess lemma is \emph{isolated} to the comb block: the open core is
$F_{\mathrm{comb}}=O(1)$, a signed-cancellation statement on the $O(p_2)$ comb generators ($p_2$ the second
prime factor), with content only once $\omega(m)\ge3$---the sup-side cancellation left open above.

\begin{conjecture}[the excess dichotomy; sup-side cancellation core]\label{conj:excess}
As $\omega(m)\to\infty$ the excess obeys exactly one of two laws, and which one is open:
\begin{itemize}
\item[\textup{(O1)}] \emph{bounded}---there is an absolute constant $C$ with $A_N-L_{\mathrm{pre}}\le C$ for
every $N$, equivalently the sharp deficit law $U_N-A_N=\tfrac1\pi\ln(m/\varphi(m))+O(1)$ of
Proposition~\ref{prop:defect-cond};
\item[\textup{(L)}] \emph{logarithmic}---$A_N-L_{\mathrm{pre}}=c\,(U_N-L_{\mathrm{pre}})+o(U_N-L_{\mathrm{pre}})$
for a constant $c\in(0,1)$, whence the sharp deficit constant is $(1-c)/\pi$ rather than $1/\pi$.
\end{itemize}
For $3\mid N$ the vertex-disjoint split reduces both to the comb block alone: \textup{(O1)} holds iff
$F_{\mathrm{comb}}(\lambda)\le C'$ for some $\lambda\in[0,1)$---a \emph{primal} (sup-side) analogue of the
McGehee--Pigno--Smith/Konyagin $L^1$ lower bound~\cite{MPS81,Konyagin81} on the $O(p_2)$ comb generators.
\end{conjecture}
\noindent The reduction is complete: Lemma~\ref{lem:triplet} bounds each clean relation, the vertex-disjoint
split isolates the triplet block (an $O(1)$ Riemann sum) from the comb block, and the prefix-budget trade-off
$\sup_\psi[\,\cdot\,]$ is the only remaining ingredient. Alternative \textup{(O1)} holds outright for
prime-power odd part (Remark~\ref{rem:defect}) and for square-free $N=3p$ (above, $C=\tfrac{\ln3}{4\pi}+o(1)$);
which alternative governs the general case is open, with content only once $\omega(m)\ge3$, and the numerical
evidence is genuinely \emph{ambiguous}---the two signals point opposite ways and neither turns over by
$\omega(m)=4$. For \textup{(O1)}: the raw excess is small ($A_N-L_{\mathrm{pre}}\in[0.05,0.12]$ through
$\omega(m)=4$) and the rigorous split upper bound
$\min_\lambda[\tfrac1{1-\lambda}\sum_{c\in T}\tfrac{b_c}2+F_{\mathrm{comb}}(\lambda)]$ stays $\approx0.19$ to
$N=1155$ while $U_N-L_{\mathrm{pre}}$ grows. For \textup{(L)}: the \emph{ratio}
$(A_N-L_{\mathrm{pre}})/(U_N-L_{\mathrm{pre}})\approx0.31\approx1/\pi$ is \emph{stable} across $\omega(m)=2,3,4$,
whereas \textup{(O1)} would force it to $0$. None of these can be pushed to a decision: since
$U_N-L_{\mathrm{pre}}=\tfrac1\pi\ln(m/\varphi(m))\sim\tfrac1\pi\ln\ln\ln N$ grows glacially ($0.28$ at
$\omega(m)=4$, only $0.31$ at $\omega(m)=6$, $N=510510$), \emph{no} computationally accessible $N$ reaches the
crossover---and the direct time-domain probe $\sup_t\sum_r b_r\sin(\omega_r t)$ is itself defeated by the
finite-time barrier for $\omega(m)\ge3$ (the recurrence floor $U_N-\sup_{t\le T}$ exceeds the excess
sixfold already at $N=105$). The dichotomy is thus intrinsically analytic. An exhaustive search of the
McGehee--Pigno--Smith/Konyagin $L^1$ circle~\cite{MPS81,Konyagin81} and its modern descendants, the
Conway--Jones/Lam--Leung vanishing-sum classification~\cite{CJ}, and cyclotomic-lattice \textsc{cvp} turns up
\emph{no} sup-side upper-bound tool---each is $L^1$/lower-bound or carries no metric---so the constrained sup
is, at present, untooled. We have ruled out the obvious levers for \textup{(O1)} (a diagonal/packing
relaxation, the local quadratic Gram form, a support-$k$ extension of Lemma~\ref{lem:triplet}, an
Ablowitz--Ladik-type conserved functional, and the covering-radius transference bound), each lossy for the
same reason: it discards the phase/sign coherence that a genuine sup-side cancellation would supply.

The family limits themselves can be pinned precisely: within each family the node-$0$ defect converges
geometrically, $d_a=\mathrm{defect}_\infty(m)-c_12^{-a}-c_24^{-a}$ (two-term fits with residuals
$\sim5\cdot10^{-9}$, the optima polished by Newton iteration on the stationarity system), giving the first
precise values of the family constants (Table~\ref{tab:famlim}). A \textsc{pslq}/inverse-symbolic search
over natural logarithmic and radical bases finds no closed form: like $\beta_{\mathrm{odd}}$, they appear
to be new constants (\texttt{verify\_defect\_family\_limits.py}). The ratio
$\mathrm{defect}_\infty(m)\big/\bigl[\tfrac1\pi\ln\tfrac m{\varphi(m)}\bigr]$ is \emph{not} universal
($0.80,\ 0.54,\ \approx0.9$), so the $O(1)$ in Proposition~\ref{prop:defect-cond} genuinely depends on the
odd part.

\begin{example}[the whole mechanism at $N=15$]\label{ex:fifteen}
The seven frequencies are $\omega_r=2\sin(\pi r/15)$, $r=1,\dots,7$ (with $\omega_5=2\sin\tfrac\pi3=\sqrt3$).
Here $\varphi(2N)=\varphi(30)=8$, so the rank is $\tfrac12\varphi(30)=4$: by Lemma~\ref{lem:prefix} the
prefix $\omega_1,\dots,\omega_4$ is $\Q$-independent and \emph{no} relation can live in indices $\le4$.
The first relation appears at index $5=\tfrac12\varphi(30)+1$---exactly the palindromic bound, here
sharp---namely
\[
  \omega_4+\omega_5=\omega_1+2\omega_2+\omega_3,\qquad\text{i.e.}\quad k=(-1,-2,-1,1,1,0,0).
\]
Its coefficient sum $\sum_r k_r=-2\equiv2\pmod4$ is nonzero mod $4$, so by the criterion of
Theorem~\ref{thm:ceiling} the target $(\tfrac\pi2,\dots,\tfrac\pi2)$ is off the subtorus and
$A_{15}<U_{15}$: composite $15$ does not saturate (Theorem~\ref{thm:classification}). Numerically
$U_{15}=0.902$, $A_{15}=0.754$, a defect $0.148$---the largest among the composites $N\le24$ because the odd
part $m=15$ carries two prime factors (Remark~\ref{rem:defect}). The lower bound of Theorem~\ref{thm:order}
aligns the four-mode prefix: $A_{15}\ge2L_{\mathrm{pre}}-U_{15}=0.47$, comfortably inside the band
$A_N\sim\tfrac1\pi\ln N$.
\end{example}

\subsection{The Dirichlet segment}\label{sub:segment}
The segment chain $\mathrm{tridiag}(-1,2,-1)$ with Dirichlet ends has the \emph{simple} spectrum
$\mu_k=2\sin\frac{\pi k}{2(N+1)}$ ($k=1,\dots,N$): no degeneracy, the structural advantage over the
ring, whose twofold $\mu_r=\mu_{N-r}$ halves the independent phases. Writing $\mu_k=2\sin(2\pi k/M)$ with
$M=4(N+1)$, the frequencies lie in $\Q(\zeta_M)$ with $2\mathrm{i}\sin(2\pi k/M)=\zeta_M^k-\zeta_M^{-k}$,
and the same minus-eigenspace argument identifies the rank \emph{exactly}. The $\mu_k$ are the first $N$
distinct frequencies of the ring on $N'=2(N+1)$ nodes; the excluded middle index $N'/2=N+1$ carries the
rational frequency $\mu_{N+1}=2$, and Theorem~\ref{thm:qrank} applied to $N'$ gives
$\dim_\Q\operatorname{span}\{\mu_1,\dots,\mu_N,2\}=\tfrac12\varphi(4(N+1))$.

\begin{theorem}[segment rank and classification]\label{thm:segrank}
\begin{equation}\label{eq:segrank}
  \dim_\Q\operatorname{span}\{\mu_1,\dots,\mu_N\}=\min\!\bigl(N,\ \tfrac12\varphi(4(N+1))\bigr),
\end{equation}
and $A_N^{\mathrm{Dir}}=U_N^{\mathrm{Dir}}$ if and only if $N+1$ is prime or a power of two.
\end{theorem}

\begin{proof}
Each $2\mathrm i\mu_k=\zeta_M^k-\zeta_M^{-k}\in V^-$ ($M=4(N+1)$), so $\dim_\Q\operatorname{span}\{\mu_k\}\le\tfrac12\varphi(M)$;
since adjoining the rational $2$ keeps the rank at $\tfrac12\varphi(M)$, the dimension equals
$\tfrac12\varphi(M)$ or $\tfrac12\varphi(M)-1$ according as $2\in\operatorname{span}\{\mu_k\}$ or not.
If $N+1=2^m$, the ring on $N'=2^{m+1}$ is $\Q$-independent (Theorem~\ref{thm:two}); then $2$ is independent
of the $\mu_k$, so $2\notin\operatorname{span}\{\mu_k\}$ and the rank is $\tfrac12\varphi(M)-1=N$. If instead
$N+1$ has an odd prime factor $p$, put $a=N+1$; then
$\sum_{j=0}^{p-1}\zeta_M^{\,a+(M/p)j}=\zeta_M^{\,a}\sum_j\bigl(\zeta_M^{M/p}\bigr)^{j}=0$, and the only term of
its imaginary part reducing to the middle index $N+1$ is $j=0$ (another would need $(M/p)j\equiv M/2$, i.e.\
$2j=p$, impossible for $p$ odd), so $\mu_{N+1}=2$ occurs with coefficient $\pm1$ and lies in
$\operatorname{span}\{\mu_1,\dots,\mu_N\}$; the rank is $\tfrac12\varphi(M)$. An elementary count gives
$\tfrac12\varphi(4(N+1))=2^a\varphi(m)$ for $N+1=2^am$ ($m$ odd), which exceeds $N$ exactly for $m=1$
($N+1=2^m$) and is $\le N$ otherwise (with equality iff $N+1$ is prime); so the rank is
$\min(N,\tfrac12\varphi(4(N+1)))$, full ($=N$) iff $N+1$ is prime or a power of two. Full rank gives
$A_N^{\mathrm{Dir}}=U_N^{\mathrm{Dir}}$ by Bohr; for composite $N+1$ the lattice $\Lambda^{\mathrm{Dir}}$ is
nontrivial and carries a relation with $\sum_r k_r\not\equiv0\ (\mathrm{mod}\ 4)$, so
$A_N^{\mathrm{Dir}}<U_N^{\mathrm{Dir}}$ by the segment ceiling criterion
(Corollary~\ref{cor:segceiling}; verified $N\le40$, \texttt{exact/gap/segment\_mod4.g},
\texttt{verify\_segment\_classification.py}; the prime case also recovers Proposition~\ref{prop:2p}). \end{proof}

\begin{corollary}[segment ceiling criterion]\label{cor:segceiling}
With $\Lambda^{\mathrm{Dir}}=\{k\in\Z^N:\sum_r k_r\mu_r=0\}$ the relation lattice of the segment frequencies,
\[
  A_N^{\mathrm{Dir}}=U_N^{\mathrm{Dir}}\quad\Longleftrightarrow\quad
  \sum_r k_r\equiv0\ (\mathrm{mod}\ 4)\ \text{ for every }k\in\Lambda^{\mathrm{Dir}}.
\]
\end{corollary}
\begin{proof}
$U_N^{\mathrm{Dir}}=\sum_r b_r$ is the $\ell^1$ mass, realised iff the $\sin(\mu_r t)$ are brought
simultaneously close to $1$, i.e.\ the target $\varphi_\ast=(\tfrac\pi2,\dots,\tfrac\pi2)$---the maximiser of
$\sum_r b_r\sin\varphi_r$---lies in the orbit closure $\mathbb T_N^{\mathrm{Dir}}$. By Kronecker,
$\varphi_\ast\in\mathbb T_N^{\mathrm{Dir}}$ iff $\langle k,\varphi_\ast\rangle=\tfrac\pi2\sum_r k_r\equiv0\
(2\pi)$, i.e.\ $\sum_r k_r\equiv0\ (4)$, for every $k\in\Lambda^{\mathrm{Dir}}$. The simple Dirichlet
spectrum has no $\mu_r=\mu_{N-r}$ degeneracy, so---unlike the ring's even-$N$ branch
(Theorem~\ref{thm:ceiling})---no antipodal node contributes; the single node-$0$ condition decides. As on
the ring (Theorem~\ref{thm:ceiling}, Remark~\ref{rem:f92}), full independence makes the criterion vacuous,
and for composite $N+1$ the lattice carries a violating relation ($\sum_r k_r\not\equiv0$ mod $4$), verified
to $N\le40$ in \texttt{exact/gap/segment\_mod4.g}.
\end{proof}

\noindent This is the exact analogue of the ring classification (Theorem~\ref{thm:classification}), shifted
by the boundary (the ring uses modulus $2N$, the segment $4(N+1)$). The order Theorem~\ref{thm:order}
transfers verbatim: Lemma~\ref{lem:prefix} holds with modulus $4(N+1)$ in place of $2N$ (same
palindromic argument), so the independent prefix has length $\tfrac12\varphi(4(N+1))$ and
$A_N^{\mathrm{Dir}}\sim\tfrac1\pi\ln N$ for \emph{every} $N$ (\texttt{verify\_segment\_lower\_bound.py}).

\section{Exact \texorpdfstring{$A_N$}{A\_N} for composite \texorpdfstring{$N$}{N}}\label{sec:exact}

For composite $N$ the frequencies satisfy rational relations and the orbit closure is a proper
\emph{subtorus} (Weyl). In the cyclotomic coordinates $C$ (row $r$ = integer coordinates of
$2\mathrm{i}\sin(\pi r/N)$ in $\Q(\zeta_{2N})$) the attainable phases are $\{\varphi=C\psi\}$, whence
\begin{equation}\label{eq:subtorus}
  \sup_t u_j(t)=\max_{\psi\in\R^p}\sum_r a_{rj}\sin\big((C\psi)_r\big),\qquad p=\varphi(2N),
\end{equation}
a finite-dimensional trigonometric maximization, reduced exactly and then evaluated numerically
(\texttt{verify\_exact\_AN.py}, returning $A_N=U_N$ on primes; a certified global bracket is given for
$N=6$ below). This makes the composite case computable for each $N$:
numerically the subtorus defect on the ring is $A_N/U_N\in[0.84,0.92]$ in the tested range, so
$A_N/\ln N\in[0.28,0.31]$, consistent with Theorem~\ref{thm:order} ($A_N\sim\tfrac1\pi\ln N$).

\begin{proposition}[algebraic composite cases]\label{prop:closed}
$A_N$ is an algebraic number: writing $x_r=\cos\varphi_r$, $y_r=\sin\varphi_r$ with $x_r^2+y_r^2=1$ and
adjoining, for each generator $k$ of the relation lattice $\Lambda$, the constraints
$\cos\langle k,\varphi\rangle=1$, $\sin\langle k,\varphi\rangle=0$ that cut out the orbit-closure subtorus
$\mathbb{T}_N$ (polynomials in the $x_r,y_r$ through the Chebyshev/addition formulas), turns
\eqref{eq:subtorus} into the maximum of a polynomial over a compact semialgebraic set defined over $\Q$,
whose value is algebraic by the Tarski--Seidenberg theorem (real quantifier elimination). For $N=6$ the relation lattice is one-dimensional and
\[
  A_6=\frac{\sqrt3}{9}+\frac{(3+\sqrt3)\,\sqrt[4]{3}}{12\sqrt2}\approx0.55942 ,
\]
with the inner maximum at $\cos\varphi=(\sqrt3-1)/2$. For $N=12$ the three-term relation
$\omega_5=\omega_1+\omega_3$ (i.e. $\sin75^\circ-\sin15^\circ=\sin45^\circ$) makes the block
two-dimensional; the algebraic degree grows with the arithmetic of $N$, and no uniform elementary
closed form is apparent.
\end{proposition}

\medskip\noindent\textbf{Worked example ($N=12$).} The six distinct frequencies $\omega_r=2\sin(\pi r/12)$
are
\[
  (\omega_1,\dots,\omega_6)=\bigl(2\sin15^\circ,\ 1,\ \sqrt2,\ \sqrt3,\ 2\sin75^\circ,\ 2\bigr)
  \approx(0.518,\ 1,\ 1.414,\ 1.732,\ 1.932,\ 2),
\]
and their relation lattice $\Lambda_{12}$ is rank two, generated by
\[
  \omega_5=\omega_1+\omega_3\quad(2\sin75^\circ-2\sin15^\circ=\sqrt2),\qquad
  \omega_6=2\omega_2\quad(2=2\cdot1).
\]
Both generators have coefficient-sum $\not\equiv0\ (\mathrm{mod}\ 4)$---reading
$\omega_5-\omega_1-\omega_3=0$ as $-1-1+1\equiv3$ and $2\omega_2-\omega_6=0$ as $2-1\equiv1$---so by the
criterion below (Theorem~\ref{thm:ceiling}) the ceiling is \emph{not} attained, $A_{12}<U_{12}$. Solving
the subtorus maximization~\eqref{eq:subtorus} numerically gives
\[
  U_{12}=\tfrac1{24}\sum_{r=1}^{11}\csc\tfrac{\pi r}{12}=0.8307,\qquad A_{12}=0.7546,\qquad
  \frac{A_{12}}{U_{12}}=0.908 ,
\]
the two relations pulling the optimal phases off $\varphi_r=\tfrac\pi2$ by just enough to cost $9\%$ of the
ceiling (\texttt{verify\_relation\_lattices.py}, \texttt{verify\_exact\_AN.py}).

A sharp arithmetic criterion decides exactly when $A_N=U_N$ (the supremum equalling the ceiling).

\begin{theorem}[ceiling criterion]\label{thm:ceiling}
Let $\Lambda=\{k\in\Z^m:\sum_r k_r\omega_r=0\}$ be the relation lattice of the distinct positive
frequencies $\omega_r=2\sin(\pi r/N)$, $1\le r\le m=\lfloor N/2\rfloor$. Then $A_N=U_N$ if and only if
\[
  \sum_{r}k_r\equiv0\ (\mathrm{mod}\ 4)\quad\text{for every }k\in\Lambda
\]
or, when $N$ is even, the antipodal variant $\sum_r k_r+2\sum_r rk_r\equiv0\ (\mathrm{mod}\ 4)$ holds for
every $k\in\Lambda$. The two conditions coincide for all $N\le160$ (where only $N$ prime and $N=2^m$
saturate, Corollary~\ref{cor:density}).
\end{theorem}

\begin{proof}
Always $A_N\le U_N$, with $U_N=\sum_r a_{r0}$ the node-$0$ ceiling. As $a_{rj}=a_{r0}\cos(2\pi rj/N)$ we
have $|a_{rj}|\le a_{r0}$, so no node exceeds $U_N$, and $|a_{rj}|=a_{r0}$ for \emph{all} $r$ forces
$|\cos(2\pi rj/N)|=1$, i.e.\ $2j/N\in\Z$: only $j=0$ and---for even $N$---$j=N/2$ can have time-supremum $U_N$.
At $j=0$ the coefficients $a_{r0}>0$, so $\sup_t G_N(0,t)=U_N$ iff all $\sin(\omega_r t)$ can be brought
simultaneously arbitrarily close to $1$, i.e.\ the target $\varphi_\ast=(\tfrac\pi2,\dots,\tfrac\pi2)$ lies in the orbit closure
$\mathbb{T}_N$. By Kronecker, $\varphi_\ast\in\mathbb{T}_N$ iff $\langle k,\varphi_\ast\rangle=
\tfrac\pi2\sum_r k_r\equiv0\ (2\pi)$ for every $k\in\Lambda$, i.e.\ $\sum_r k_r\equiv0\ (4)$. At $j=N/2$
(even $N$) the signs $\cos(\pi r)=(-1)^r$ give the alternating target $\varphi_r=\tfrac\pi2+\pi r$, and the
same argument yields $\langle k,\varphi\rangle=\tfrac\pi2\sum_r k_r+\pi\sum_r rk_r\equiv0\ (2\pi)$, i.e.\
$\sum_r k_r+2\sum_r rk_r\equiv0\ (4)$. Thus $A_N=U_N$ iff one of the two admissible nodes can realize
its target, giving the stated disjunction (\texttt{verify\_ceiling\_criterion.py}).
\end{proof}

\begin{remark}[priority]\label{rem:f92}
The degenerate (``multiple-frequency'') regime is the subject of Filimonov's 1992 note on the circular
string~\cite{F92} (Zbl~0761.73053), now obtained and compared. With the same frequencies
$\omega_k=2\sin(\pi k/N)$, its Theorems~1--4 establish: (Thm~1) the ceiling $\sup_t S_j/F=b_j$ under
$\Q$-independence, by Kronecker; (Thm~2) $\Q$-independence for $N$ prime or $2^m$, by a cyclotomic
minimal-polynomial argument; (Thm~3) $\Q$-\emph{dependence} for even composite $N=2^m q$ ($q$ odd $\ge3$),
via $\varphi(2N)=N-2^m<N$; and (Thm~4) the unbounded ceiling $b_j(N)\to\infty$ (with no rate). So the
$\Q$-independence dichotomy for the ring, and the ceiling attained on it, are Filimonov's. What~\cite{F92}
does \emph{not} contain, and is new here, is the $\bmod\,4$ criterion---which decides when \emph{dependent}
frequencies still saturate, hence the exact classification $A_N=U_N\Leftrightarrow N$ prime or $2^m$ (its
Thm~3 gives dependence for even composite only, and does not show that dependence \emph{lowers} the
ceiling)---together with the rate $\tfrac1\pi\ln N$, the constant, and the closed-form all-$N$ rank
$\tfrac12\varphi(2N)$.
\end{remark}

This refines the independence test: $\Q$-independence means $\Lambda=\{0\}$, which satisfies the
criterion vacuously, recovering $A_N=U_N$ for $N$ prime and $N=2^m$. The criterion is genuinely weaker in
principle---a composite $N$ with dependent frequencies could still saturate, provided every relation has
coefficient-sum divisible by $4$. Computing $\Lambda$ exactly by integer (Hermite) reduction of the
cyclotomic coordinates and testing the criterion against the directly optimized $A_N/U_N$, the two
agree for all $N\le40$, and a criterion-only scan finds \emph{no} composite $N\le160$ that saturates:
on this range $A_N=U_N$ holds precisely for $N$ prime or $N=2^m$ (\texttt{verify\_ceiling\_criterion.py}).
The \emph{full-independence} route to saturation is in fact closed for \emph{all} $N$ by
Theorem~\ref{thm:qrank} ($\operatorname{rank}=\tfrac12\varphi(2N)=\lfloor N/2\rfloor$ only for prime and
$2^m$, confirmed numerically to $N\le500$, \texttt{verify\_order\_extended.py}); a composite saturator
would have to satisfy the mod-$4$ criterion \emph{despite} rational dependence, which none $\le160$ does.
Table~\ref{tab:lattices} makes the lattices explicit; e.g.\ the generators
$\omega_3=2\omega_1$ ($N=6$), $\omega_5=\omega_1+\omega_3$ ($N=12$, cf.\ Proposition~\ref{prop:closed}),
and $\omega_4+\omega_5=\omega_1+2\omega_2+\omega_3$ ($N=15$) all have coefficient-sum $\not\equiv0\ (4)$.

\begin{corollary}[saturating families and a finite scan]\label{cor:density}
The two proven saturating families, the primes and the powers of two, both have natural density zero.
An exhaustive criterion scan finds no further saturating $N\le160$, so on this range
$\{N:A_N=U_N\}=\{N\text{ prime}\}\cup\{2^m\}$. Theorem~\ref{thm:classification} promotes this to all $N$:
the saturating set is \emph{exactly} the primes and the powers of two, hence of density zero. This set is
OEIS~A174090~\cite{OEIS}, with the curious equivalent description: the ring saturates exactly when $N$
\emph{cannot} be written as a sum of three or more consecutive positive integers.
\end{corollary}

\begin{theorem}[classification of saturation]\label{thm:classification}
$A_N=U_N$ if and only if $N$ is prime or a power of two.
\end{theorem}

\begin{proof}
If $N$ is prime or $N=2^m$ then $\Lambda_N=\{0\}$ (Theorems~\ref{thm:prime},~\ref{thm:two}) and
Theorem~\ref{thm:ceiling} gives $A_N=U_N$. Conversely, let $N$ be composite and not a power of two, and
let $p$ be an odd prime factor, so $N=ps$ with $s=N/p\ge2$ and $q:=2N/p=2s$. As
$\zeta_{2N}^{\,q}=\eu^{2\pi\mathrm i/p}$ is a primitive $p$th root of unity,
\[
  \sum_{j=0}^{p-1}\zeta_{2N}^{\,1+qj}=\zeta_{2N}\sum_{j=0}^{p-1}\bigl(\zeta_{2N}^{\,q}\bigr)^{j}=\zeta_{2N}\cdot0=0,
\]
and taking imaginary parts, $\sum_{j=0}^{p-1}\sin\frac{\pi(1+qj)}{N}=0$. Reduce each term to a distinct
frequency: with $e_j:=(1+qj)\bmod 2N$ one has $\sin\frac{\pi(1+qj)}{N}=\sigma_j\sin\frac{\pi r_j}{N}$, where
$r_j\in\{1,\dots,\lfloor N/2\rfloor\}$ and $\sigma_j=+1$ if $e_j\in(0,N)$, $\sigma_j=-1$ if $e_j\in(N,2N)$.
\emph{No $e_j$ lies in $\{0,N\}$:} $e_j\equiv0$ would need $qj\equiv-1\ (2N)$, impossible since $q\mid 2N$
and $q>1$; $e_j\equiv N$ would need $q\mid N-1$, but $p$ odd gives $ps\equiv s\ (2s)$, hence
$N-1\equiv s-1\ (2s)$ with $0<s-1<2s$, so $2s\nmid N-1$. Thus every $\sigma_j=\pm1$, and the vector $k$
with $k_r=\sum_{j:\,r_j=r}\sigma_j$ satisfies $\sum_r k_r\,2\sin\frac{\pi r}N=\sum_j\sigma_j\,2\sin\frac{\pi r_j}N=0$,
i.e.\ $k\in\Lambda_N$, with
\[
  \sum_r k_r=\sum_{j=0}^{p-1}\sigma_j\equiv p\equiv1\pmod 2 .
\]
So $\sum_r k_r$ is odd: then $\sum_r k_r\not\equiv0\ (4)$, and $\sum_r k_r+2\sum_r rk_r$ is also odd, hence
$\not\equiv0\ (4)$. The single relation $k$ violates \emph{both} branches of Theorem~\ref{thm:ceiling}, so
neither the node $j=0$ nor (for even $N$) the node $j=N/2$ can saturate, and $A_N<U_N$
(\texttt{verify\_classification\_proof.py}). \end{proof}

\medskip\noindent\textbf{The composite gap.} For every composite $N$, then, $A_N<U_N$
(Theorem~\ref{thm:classification}). The gap is
\emph{certified} for $N=6$: the lone relation $2\omega_1=\omega_3$ has coefficient-sum
$2+0-1\equiv3\ (4)\ne0$, so by Theorem~\ref{thm:ceiling} $A_6<U_6$; quantitatively a Lipschitz-grid
certificate on the two-dimensional subtorus---the elementary level of the Lasserre moment--SOS
hierarchy~\cite{Lasserre}---brackets $A_6\in[0.5594,0.5604]$, so $U_6-A_6\ge0.048>0$, while the prime
$N=5$ returns $A_5=U_5$ (\texttt{verify\_sos\_certified.py}).

\begin{proposition}[twice a prime: the sharp constant on a composite family]\label{prop:2p}
For $N=2p$ with $p$ an odd prime, $A_N=U_N-O(1/N)=\tfrac1\pi\ln(2p)+O(1)$; the sharp constant $1/\pi$
holds, extending Theorems~\ref{thm:prime}--\ref{thm:two} to an infinite family of \emph{composite} $N$.
\end{proposition}

\begin{proof}
The prefix $\{\omega_1,\dots,\omega_{p-1}\}=\{2\sin(\pi r/2p):1\le r\le p-1\}$ is $\Q$-independent: in
$\Q(\zeta_{4p})$, where $\Phi_{4p}(x)=\Phi_p(-x^2)$ has degree $\varphi(4p)=2(p-1)$, the values
$2\mathrm i\sin(\pi r/2p)=\zeta_{4p}^{\,r}-\zeta_{4p}^{-r}$ ($1\le r\le p-1$) are linearly independent over
$\Q$. Indeed, if $\sum_{r=1}^{p-1}c_r(\zeta_{4p}^{\,r}-\zeta_{4p}^{-r})=0$ with $c_r\in\Q$, multiplying by
$\zeta_{4p}^{\,p-1}$ gives $Q(\zeta_{4p})=0$, where $Q(x)=\sum_{r=1}^{p-1}c_r(x^{p-1+r}-x^{p-1-r})$ has
degree $\le2p-2=\deg\Phi_{4p}$; as $\Phi_{4p}$ is the minimal polynomial of $\zeta_{4p}$, this forces
$Q=C\,\Phi_{4p}$, but $Q(1)=0$ while $\Phi_{4p}(1)=\Phi_p(-1)=1$, so $C=0$ and $Q\equiv0$, and since the
exponent sets $\{p-1+r\}_r$ and $\{p-1-r\}_r$ are disjoint, every $c_r=0$ (exact integer rank $=p-1$,
\texttt{verify\_2p\_family.py}). Thus the sole $\Q$-dependent frequency is the rational $\omega_p=2$, of
weight $\tfrac1{2N}$. With the degeneracy
$\omega_r=\omega_{N-r}$ the node-$0$ response is
$u_0(t)=\tfrac2N\sum_{r=1}^{p-1}\omega_r^{-1}\sin(\omega_r t)+\tfrac1{2N}\sin 2t$. By Kronecker the
independent prefix can be driven to $\sin(\omega_r t)\to1$ simultaneously, and since
$\tfrac2N\sum_{r=1}^{p-1}\omega_r^{-1}=\tfrac1N\sum_{r=1}^{p-1}\csc\tfrac{\pi r}{2p}=U_N-\tfrac1{2N}$,
we get $\sup_t u_0(t)\ge U_N-\tfrac1N$. As $A_N\le U_N$ always, $A_N\in[U_N-\tfrac1N,\,U_N]$
(\texttt{verify\_2p\_family.py}).
\end{proof}

\subsection{Weyl equidistribution and the lower-bound problem}\label{sub:weyl}
The orbit-closure optimization is rigorous via Weyl's equidistribution theorem~\cite{Weyl}. The frequency vector
$\omega=(\omega_1,\dots,\omega_m)$ defines the linear flow $t\mapsto t\omega\bmod2\pi$, whose closure is
the subtorus
\[
  \mathbb{T}_N=\bigl\{\varphi\in\mathbb{T}^{m}:\langle k,\varphi\rangle\equiv0\ (2\pi)\ \forall k\in\Lambda\bigr\},
  \qquad \Lambda=\{k\in\Z^m:\langle k,\omega\rangle=0\},
\]
of dimension $\dim\mathbb{T}_N=m-\operatorname{rank}\Lambda=\operatorname{rank}_\Q\{\omega_r\}$, the
$\Q$-rank of Section~\ref{sec:prime}. This rank has a closed form, valid for \emph{all} $N$.

\begin{theorem}[subtorus dimension]\label{thm:qrank}
For every $N\ge3$,
\begin{equation}\label{eq:qrank}
  \dim\mathbb{T}_N=\operatorname{rank}_\Q\{2\sin(\tfrac{\pi r}{N}):1\le r\le N-1\}=\tfrac12\varphi(2N).
\end{equation}
\end{theorem}

\begin{proof}
Write $\zeta=\zeta_{2N}$ and let $\sigma:\zeta\mapsto\zeta^{-1}$ be complex conjugation, an involution of
$\Q(\zeta)$ whose fixed field is the maximal real subfield $\Q(\zeta+\zeta^{-1})$, of degree
$\tfrac12\varphi(2N)$. In characteristic $0$, $\Q(\zeta)=V^+\oplus V^-$ where $V^\pm$ are the
$\pm1$-eigenspaces of $\sigma$; thus $\dim_\Q V^\pm=\tfrac12\varphi(2N)$, and since $\sigma$ is the
restriction of conjugation, $V^-$ is exactly the set of purely imaginary elements of $\Q(\zeta)$.
Now $2\mathrm i\sin(\pi r/N)=\zeta^r-\zeta^{-r}=(1-\sigma)\zeta^r$. For any $y$, $\sigma((1-\sigma)y)=
-(1-\sigma)y$, so $(1-\sigma)\Q(\zeta)\subseteq V^-$; and $(1-\sigma)x=2x$ for $x\in V^-$, so the inclusion
is an equality. The powers $\{\zeta^r:r\in\Z\}$ span $\Q(\zeta)$ over $\Q$ (they contain the power basis
$1,\zeta,\dots,\zeta^{\varphi(2N)-1}$), hence their images $\{\zeta^r-\zeta^{-r}\}$ span
$(1-\sigma)\Q(\zeta)=V^-$. Finally $\zeta^N=-1$ gives $\zeta^{N+s}-\zeta^{-(N+s)}=-(\zeta^s-\zeta^{-s})$
while $r=0,N$ give $0$, so the values for $1\le r\le N-1$ already span $V^-$. Dividing by $2\mathrm i$ is a
$\Q$-linear bijection of $V^-\subset\mathrm i\R$ onto $\operatorname{span}_\Q\{\sin(\pi r/N)\}\subset\R$,
which therefore has dimension $\dim_\Q V^-=\tfrac12\varphi(2N)$.
\end{proof}

\noindent The formula is independently confirmed by an exact integer-rank certificate for $N\le64$
(\texttt{verify\_qrank\_formula.py}); the dimension sequence $\tfrac12\varphi(2N)$ is
OEIS~A055034~\cite{OEIS}. Hence full rank---and therefore $A_N=U_N$---holds \emph{exactly}
when $\tfrac12\varphi(2N)=\lfloor N/2\rfloor$, i.e.\ for $N$ prime or $N=2^m$ (Figure~\ref{fig:qrank}):
precisely the scope of Theorems~\ref{thm:prime} and~\ref{thm:two}, now the full-rank case of one
formula. Hence \eqref{eq:subtorus} is \emph{exactly}
\[
  A_N=\max_j\ \max_{\varphi\in\mathbb{T}_N}\frac2N\sum_r\cos\frac{2\pi rj}{N}\,\sin\varphi_r .
\]
For any $S$ with
$\{\omega_r\}_{r\in S}$ $\Q$-independent, the projection of $\mathbb{T}_N$ onto the $S$-coordinates is
the full torus $\mathbb{T}^{|S|}$, so those phases align and contribute $(2/N)\sum_{r\in S}\omega_r^{-1}$.
Taking $S=\{1,\dots,M(N)\}$ the independent prefix, $M(N)=\tfrac12\varphi(2N)$ by Lemma~\ref{lem:prefix},
this is the budget $L_{\mathrm{pre}}$ of the order Theorem~\ref{thm:order}.

\smallskip
\noindent\textbf{The lower bound, settled.} The earlier obstruction---that the out-of-prefix modes might
interfere destructively---is dispatched by the crude bound $\sin\ge-1$: it costs at most
$U_N-L_{\mathrm{pre}}$, and because Lemma~\ref{lem:prefix} makes the prefix \emph{long}
($M(N)=\tfrac12\varphi(2N)$, within a $\ln\ln N$ factor of $N/2$), the surviving amount
$2L_{\mathrm{pre}}-U_N=(1-o(1))U_N$ still grows like $\tfrac1\pi\ln N$. \emph{Soft} bounds had failed here
for a structural reason---the time-averaged second moment gives only $\mathrm{RMS}_t\to1/\sqrt{12}$ and the
slowest single mode only $b_1\to1/\pi$, both $O(1)$---so the $\ln N$ genuinely needed the simultaneous
alignment of $\Theta(\ln N)$ \emph{arithmetically independent} modes, which Lemma~\ref{lem:prefix} supplies.
Table~\ref{tab:order} shows the resulting band: the exact $A_N/\ln N$ stays in $[0.28,0.34]$ across all
classes (including the hardest $N=pq$, $p,q\sim\sqrt N$, where $A_N/U_N\ge0.98$), consistent with
Theorem~\ref{thm:order}.

\section{Reachability of the ceiling: finite time and disorder}\label{sec:reach}

The supremum $A_N$ is an \emph{infinite-time} quantity. Two refinements bring out its physical content:
how long one must wait, and how fragile the arithmetic that makes the wait short.

\subsection{Finite-time amplification}
Define the finite-horizon amplitude
\[
  A_N(T)=\sup_{0\le t\le T}\,\max_j|u_j(t)|\ \le\ A_N .
\]
Reaching
$(1-\varepsilon)U_N$ requires $\sin(\omega_r t)\approx1$ for all $m=\lfloor N/2\rfloor$ frequencies at
once---a simultaneous Diophantine approximation of the phases to $\tfrac\pi2$. By Dirichlet, such an
alignment to tolerance $\delta$ \emph{exists} by time $\sim\delta^{-m}$---a crude exponential
\emph{upper} bound on the recurrence time
\[
  T_\varepsilon(N)=\inf\{T:A_N(T)\ge(1-\varepsilon)U_N\}.
\]
(This targets $U_N$, so $T_\varepsilon(N)$ is finite only on the saturating classes $N$ prime and $2^m$;
for non-saturating $N$, where $A_N<U_N$, one replaces $U_N$ by $A_N$.) A matching \emph{lower} bound
(that one must wait at least exponentially long) does \emph{not} follow
from Dirichlet, which bounds the wait from above; it would require Diophantine lower bounds on the
badly-approximable phase vector, and is left open. Numerically the wait does grow exponentially: over primes $11\le N\le47$, $\log T_{0.2}(N)$ has slope
$\approx0.25$ per unit $N$, and across $\varepsilon$ the recurrence time fits the simultaneous-approximation
law $T_\varepsilon(N)\sim\varepsilon^{-a m}$ with $a\approx0.32$, $m=\lfloor N/2\rfloor$ ($R^2=0.88$ up to
$N\le71$; \texttt{verify\_finite\_time\_lower\_bound.py}, GPU, Figure~\ref{fig:reach})---the steeper
$\approx0.63$ slope on $N\le19$ alone is pre-asymptotic, inflated by the $O(1)$ times at $N=5,7$. For small
primes the \emph{upper} bound is moreover certified rigorously---at an explicit witness $\tilde t$ the
amplitude exceeds $0.8\,U_N$ in interval arithmetic, so $T_{0.2}(N)\le\tilde t$
($\tilde t\approx0.69,\,0.83,\,147,\,333$ for $N=5,7,11,13$; \texttt{verify\_finite\_time\_certified.py}). At
a \emph{fixed} horizon $T$ the efficiency $A_N(T)/U_N$ \emph{decreases} with $N$ (at $T=10^3$, from
$\approx1.0$ at $N=5$ to $\approx0.58$ at $N=71$): more frequencies are harder to align in bounded time. Thus the logarithmic law is an infinite-time ceiling; a physical large wave is the
partial early alignment, already a several-fold over-amplification, while the full $U_N$ is approached
only over astronomically long times. The clean arithmetic of prime and $2^m$ $N$ is what makes a sizable
fraction of $U_N$ reachable \emph{quickly}.

\subsection{Disorder}
Unequal masses $m_j=1+\varepsilon\xi_j$ turn the chain into the generalized problem $\LN v=\omega^2 Mv$,
$M=\operatorname{diag}(m_j)$, with node-$0$ response $x_0(t)=\sum_r c_r\sin(\omega_r t)$,
$c_r=v_r(0)^2/\omega_r>0$ ($M$-orthonormal modes $v_r^\top M v_s=\delta_{rs}$, unit velocity impulse) and
incoherent ceiling $C=\sum_r c_r$. Generic masses destroy every rational relation almost surely. Each
$F_k(\xi)=\sum_r k_r\omega_r(\xi)$ ($k\in\Z^{N-1}\!\setminus\!\{0\}$) is real-analytic on the simple-spectrum
set; it is not \emph{identically} zero because the eigenvalue map $\xi\mapsto(\lambda_r)$ is a submersion---its
Jacobian $\partial\lambda_r/\partial m_j=-\lambda_r v_r(j)^2$ has full row rank $N-1$ at generic masses
(\texttt{verify\_disorder.py}; full rank is an open condition, certifiable at a witness), so the
$\omega_r=\sqrt{\lambda_r}$ vary independently and no nontrivial linear combination can vanish on an open set.
Hence each $\{F_k=0\}$ is a proper analytic subset, of measure zero, and their countable union still has
measure zero; so the obstruction of Theorem~\ref{thm:ceiling} disappears and \emph{every} $N$ can in principle
reach its ceiling. We verify (\texttt{verify\_disorder.py}) that the exact resonance
residual of a composite $N$ jumps from $0$ to $O(\varepsilon)$ under disorder, and that over a long
horizon the achievable efficiency $A/C$ rises above the blocked ordered value (e.g.\ $N=15$:
$0.83\!\to\!0.91$). But the gain is purely infinite-time: at a short horizon the same disorder yields no
improvement, because the now-independent but near-resonant frequencies need Diophantine time to align.
The dichotomy is sharp: an infinitesimal perturbation makes the absolute ceiling attainable in principle
while pushing the time to attain it beyond any physical horizon. Infinite-time amplification improves;
finite-time amplification does not. The next lemma makes this dichotomy quantitative: at a fixed horizon the
disordered response is Lipschitz in $\varepsilon$, so a gain forces the horizon to grow like $1/\varepsilon$.

\begin{lemma}[finite horizon is Lipschitz in disorder; a gain needs $T\gtrsim1/\varepsilon$]\label{lem:disorder-finiteT}
Fix $N$ and a direction $\xi$ with $\|\xi\|_\infty\le1$, and let $x_0^{(\varepsilon)}(t)=\sum_r c_r(\varepsilon)\sin(\omega_r(\varepsilon)t)$
be the node-$0$ response at masses $m_j=1+\varepsilon\xi_j$. There is $C_N>0$ (depending only on $N$) with
\[
  \Bigl|\sup_{0\le t\le T}x_0^{(\varepsilon)}(t)-\sup_{0\le t\le T}x_0^{(0)}(t)\Bigr|\le C_N\,(1+T)\,\varepsilon,
  \qquad 0\le\varepsilon\le\tfrac12 .
\]
Hence a fixed gain $\delta>0$ over the ordered ring needs $\varepsilon\ge\delta/\bigl(C_N(1+T)\bigr)$: at fixed $T$ an
infinitesimal disorder gives no improvement, and to exploit disorder the horizon must grow as $T\gtrsim\delta/(C_N\varepsilon)$.
\end{lemma}
\begin{proof}
Write $B(\varepsilon)=M^{-1/2}\LN M^{-1/2}$ ($M=\operatorname{diag}(1+\varepsilon\xi_j)$) and $g_t(\lambda)=\sin(\sqrt\lambda\,t)/\sqrt\lambda$
on the oscillatory subspace $\lambda>0$; the zero mode is structural ($\LN\mathbf1=0$ for every mass profile, eigenvector
$\mathbf1$) and contributes no oscillation. With $M$-normalized modes one has the closed form
$x_0^{(\varepsilon)}(t)=m_0^{-1}\,[g_t(B(\varepsilon))]_{00}$. For $\|\xi\|_\infty\le1,\ \varepsilon\le\tfrac12$ the family
$B(\varepsilon)$ is real-analytic with $\|\partial_\varepsilon B\|\le K_1$, and its nonzero spectrum stays in a fixed
$[\delta_N,4]$, $\delta_N>0$ (continuity on the compact $\varepsilon$-interval). On $[\delta_N,4]$ the symbol $g_t$ is smooth
with Lipschitz constant $\le a_N+b_N t$ (from $g_t'(\lambda)=\tfrac{t\cos\sqrt\lambda t}{2\lambda}-\tfrac{\sin\sqrt\lambda t}{2\lambda^{3/2}}$),
so $\|g_t(B(\varepsilon))-g_t(B(0))\|\le(a_N+b_N t)K_1\varepsilon$. Combined with $|m_0^{-1}-1|\le2\varepsilon$ and the
$t$-uniform bound $|[g_t(B)]_{00}|\le\delta_N^{-1/2}$, this gives $|x_0^{(\varepsilon)}(t)-x_0^{(0)}(t)|\le C_N(1+t)\varepsilon$.
The supremum over $t\le T$ is $1$-Lipschitz in the sup-norm, and the contrapositive is the stated horizon requirement.
\end{proof}

\section{The discrete Schr\"odinger equation on the ring}\label{sec:schr}

Following Filimonov~\cite{F23}, consider $\mathrm{i}\dot z=\LN z$, so $z(t)=\eu^{-\mathrm{i}t\LN}z(0)$ with kernel
\[
  K_N(k,t)=\frac1N\sum_r\eu^{-\mathrm{i}t\lambda_r}\,\eu^{2\pi\mathrm{i}rk/N}.
\]
Here the phases are governed by $\lambda_r=2-2\cos\tfrac{2\pi r}{N}$, not by $\omega_r=\sqrt{\lambda_r}$.

\begin{theorem}[Schr\"odinger contrast]\label{thm:schr}
\begin{enumerate}
\item (Unitarity.) For $z(0)=e_0$, $\ \max_j|z_j(t)|\le1$ for all $t$: a localized state is never
amplified, in sharp contrast with the bead chain.
\item (Ballistic $\ell^\infty$ amplification.)
$B_N:=\sup_t\|\eu^{-\mathrm{i}t\LN}\|_{\ell^\infty\to\ell^\infty}=\sup_t\sum_k|K_N(k,t)|\le\sqrt N$
(each row of the unitary $\eu^{-\mathrm it\LN}$ has unit $\ell^2$ norm, so $\ell^1$ norm $\le\sqrt N$).
Together with the matching lower bound (Theorem~\ref{thm:schr-lower}), $B_N=\Theta(\sqrt N)$, not $\ln N$.
\item (Arithmetic.) The spectrum lies in the real cyclotomic field $\Q(\zeta_N)^+$ via $\cos$ (whereas
$\{\omega_r\}\subset\Q(\zeta_{2N})$ via $\sin$); the affine identity $\sum_r\lambda_r=2N$ holds always,
and $\rank_\Q\{\lambda_r\}$ is strictly smaller than $\rank_\Q\{\omega_r\}$ for $N=2^m$
(e.g. $4$ vs $8$ at $N=16$).
\end{enumerate}
\end{theorem}

\begin{lemma}[Bessel $\ell^1$ asymptotics]\label{lem:bessel-l1}
As $x\to\infty$, $\displaystyle\sum_{n\in\Z}|J_n(x)|=c_0\sqrt x\,(1+o(1))$ with
$c_0=(2/\pi)^{3/2}B(\tfrac12,\tfrac34)\approx1.217$.
\end{lemma}

\begin{proof}
Split at the light cone $|n|=x$. Beyond it ($|n|\ge x(1+\delta)$) the bound $|J_n(x)|\le(x/2)^{|n|}/|n|!$
gives super-exponential decay; the turning-point band $|n|=x+O(x^{1/3})$ carries $O(x^{1/3})$ terms of
size $O(x^{-1/3})$, contributing $O(1)$; and the shell $(1-\delta)x\le|n|\le x$ contributes, by the
integrable amplitude bound below, $O(\delta^{3/4}\sqrt x)$---all $o(\sqrt x)$ after $\delta\to0$. In the
oscillatory region $|n|\le(1-\delta)x$ the Debye expansion~\cite{Watson} holds uniformly,
\[
  J_n(x)=\sqrt{\tfrac{2}{\pi\sqrt{x^2-n^2}}}\,\bigl(\cos\Phi_n+O(x^{-1})\bigr),\qquad
  \Phi_n=\sqrt{x^2-n^2}-n\arccos\tfrac nx-\tfrac\pi4 ,
\]
and the phase increments $\Phi_{n+1}-\Phi_n=-\arccos(n/x)$ vary with $n$, so the $\{\Phi_n\}$
equidistribute and $|\cos\Phi_n|$ averages to $\langle|\cos|\rangle=2/\pi$ (van der Corput). Hence
\[
  \sum_{|n|\le(1-\delta)x}|J_n(x)|=(1+o(1))\,\frac2\pi\sqrt{\tfrac2\pi}\int_{-x}^{x}\frac{dn}{(x^2-n^2)^{1/4}}
  =(1+o(1))\,(2/\pi)^{3/2}B(\tfrac12,\tfrac34)\,\sqrt x ,
\]
the integral equalling $\sqrt x\,B(\tfrac12,\tfrac34)$ (set $n=x\sin\theta$). Letting $\delta\to0$ gives the
claim; the leading constant is confirmed to the digits of Table~\ref{tab:bn}.
\end{proof}

\begin{theorem}[Schr\"odinger lower bound]\label{thm:schr-lower}
$\liminf_N B_N/\sqrt N\ge c_0/\sqrt2$ with $c_0=(2/\pi)^{3/2}B(\tfrac12,\tfrac34)$, i.e.\
$\liminf_N B_N/\sqrt N\ge0.8607$. With Theorem~\ref{thm:schr}(2), $B_N=\Theta(\sqrt N)$.
\end{theorem}

\begin{proof}
Evaluate the $\ell^1$ mass at the aliasing-free time $t=\tfrac N4(1-\delta)$, so $2t=\tfrac N2(1-\delta)$.
By the Poisson periodization \eqref{eq:poisson}, $K_N(j,t)=\sum_{\ell}K_\infty(j+\ell N,t)$ with
$|K_\infty(n,t)|=|J_n(2t)|$. The Bessel front sits at $|n|\approx2t=\tfrac N2(1-\delta)$, and the nearest
aliased copy ($\ell\ne0$) lies at $|j+\ell N|\ge N-2t=\tfrac N2(1+\delta)>2t$, where the bound
$|J_n(2t)|\le t^{|n|}/|n|!$ forces super-exponential decay; so
$\sum_{j}|K_N(j,t)|=(1+o(1))\sum_{n\in\Z}|J_n(2t)|$. By Lemma~\ref{lem:bessel-l1} this is
$(1+o(1))\,c_0\sqrt{2t}=(1+o(1))\,c_0\sqrt{N/2}\,\sqrt{1-\delta}$; letting $\delta\to0$,
$\liminf_N B_N/\sqrt N\ge c_0/\sqrt2$. The upper bound $B_N\le\sqrt N$ is Theorem~\ref{thm:schr}(2)
(\texttt{verify\_bn\_parity.py}). \end{proof}

\noindent The improvement over the cruder $t=N/8$ value $\tfrac12 c_0=0.6086$ is the optimal aliasing-free
time $t=N/4$, at which the Bessel front exactly fills one half-period.

\begin{remark}[the candidate constant splits by parity]\label{rem:bn-parity}
The sharp constant is expected to be parity-dependent: the aliasing phase $\mathrm i^{-N}$ is real for even
$N$ and imaginary for odd $N$, so where the two overlapping fronts meet they add differently. Numerically
(\texttt{verify\_bn\_parity.py}) the sup is at $t=N/4$ for even $N$, giving $B_N/\sqrt N\to c_0/\sqrt2$ (the
value above), and at $t=N/2$ for odd $N$, where the two surviving aliasing copies give
$|K_N(k,N/2)|\approx\sqrt{J_k(N)^2+J_{N-k}(N)^2}$ across the front (no cancellation, since $\mathrm i^{-N}$
is imaginary). The Debye envelope $|J_{Nu}(N)|\sim\sqrt{2/\pi N}\,(1-u^2)^{-1/4}|\cos\Phi|$
(the uniform large-order asymptotics of~\cite[\S10.19]{DLMF}) then yields the
limiting value
\begin{equation}\label{eq:beta-odd}
  \beta_{\mathrm{odd}}=\sqrt{\tfrac2\pi}\int_0^1\mathbb E_{\Phi,\Psi}
     \sqrt{\frac{\cos^2\Phi}{\sqrt{1-u^2}}+\frac{\cos^2\Psi}{\sqrt{u(2-u)}}}\;\mathrm du=0.928019304807931\ldots,
\end{equation}
an elliptic-integral average ($\Phi,\Psi$ independent uniform). Computed to $25$ digits, it matches no
combination of $\pi,\eu^\gamma,\Gamma(\tfrac14),\sqrt2$, Catalan's constant, $\ldots$, and is not a low-degree
algebraic number (an integer-relation/\textsc{pslq} search returns nothing): no elementary closed form was
found, and we treat $\beta_{\mathrm{odd}}$ as a new constant defined by \eqref{eq:beta-odd}
(\texttt{verify\_beta\_odd.py}). The evaluations at $t=N/4$ and $t=N/2$ give the rigorous lower bounds
$\liminf_N B_N/\sqrt N\ge c_0/\sqrt2$ (Theorem~\ref{thm:schr-lower}) and
$\liminf_{\text{odd }N}B_N/\sqrt N\ge\beta_{\mathrm{odd}}$, whence
$\limsup_N B_N/\sqrt N\in[\beta_{\mathrm{odd}},1]$; numerically the even subsequence attains $c_0/\sqrt2$,
so the global $\liminf$ equals it.
For the upper side, on the whole window $t\le N/2$ the two-copy reduction is rigorous: $2t\le N$, so only
the two Poisson copies $\ell=0,-1$ survive (the rest are super-exponentially small), $|K_N(k,t)|\le
|J_k(2t)|+|J_{N-k}(2t)|$, and the Debye envelope turns $\|K_N(\cdot,t)\|_1/\sqrt N$ into the two-copy profile
$F(2t/N)$. This profile is now explicit and its maximizer is \emph{proved}
(Proposition~\ref{prop:bn-profile}): $\max F=F(1)=\beta_{\mathrm{odd}}$ at $t=N/2$, analytically on
$[\tfrac12,0.937]$ and by validated interval arithmetic on $[0.937,1]$. Hence
$\|K_N(\cdot,t)\|_1\le(\beta_{\mathrm{odd}}+o(1))\sqrt N$ for all
$t\le N/2$, and with the lower bound $\sup_{t\le N/2}\|K_N(\cdot,t)\|_1=(\beta_{\mathrm{odd}}+o(1))\sqrt N$
on the odd subsequence. For $t>N/2$ (three or more overlapping copies) the joint equidistribution of the
overlapping Debye phases is now \emph{proved}, with an explicit power saving
(Theorem~\ref{thm:joint-equi}, \S\ref{sub:multicopy}), and it transfers uniformly in $u$ to the $L^1$ limit
(Theorem~\ref{thm:uniform-transfer}), so the entire multi-copy constant reduces \emph{unconditionally} to the
toral profile $F(s)$ of Proposition~\ref{prop:multicopy}: $\|K_N(\cdot,sN/2)\|_1=(F(s)+o(1))\sqrt N$ for every
fixed $s>1$, sharper than the bare unitarity $\|K_N(\cdot,t)\|_1\le\sqrt N$. Numerically the $\ell^1$ mass
there falls back toward its
time average $\approx0.79\sqrt N$ and never exceeds the $t=N/2$ value (checked out to $t=5N$ at
$N=2^{20}$, where the $t>N/2$ supremum trails the $t=N/2$ value by a stable margin): the $\sin^2$
dispersion admits no flat (Gauss-sum) revival, even at structured Talbot times $t=\pi p/q$. This is consistent with the
dispersive-quantization picture---exact revivals of the periodic Schr\"odinger flow occur only for a
\emph{quadratic} dispersion (rational times give Gauss sums)~\cite{Olver}, which the discrete
$\lambda_r=4\sin^2(\pi r/N)$ is not. So numerically the global sup sits at $t=N/2$, and closing the residual
upper bound would give $\limsup_N B_N/\sqrt N=\beta_{\mathrm{odd}}$. The residual on $t>N/2$ has a clean form: $K_N(\cdot,t)$ is the inverse DFT
of the \emph{unimodular} symbol $(\eu^{-\mathrm i t\lambda_r})_r$---a chirp, since $2t\cos(2\pi r/N)$ is
locally quadratic---so $B_N/\sqrt N$ is the maximal $L^1$-flatness ratio $\|P\|_1/\|P\|_2$ of a unimodular
trigonometric polynomial. The limiting macroscopic profile of $\|K_N(\cdot,t)\|_1/\sqrt N$ is a
\emph{multi-copy} Debye average over the Poisson copies (\texttt{verify\_bn\_residual\_profile.py}: it
matches the FFT to $0.5\%$ across all $N\bmod4$, peaks at $t=N/2$ with $\beta_{\mathrm{odd}}$, and is
strictly subdominant for $t>N/2$, stably out to $t=5N$), the profile $F(s)$ of
Proposition~\ref{prop:multicopy}. The joint equidistribution $\bmod\,2\pi$ of the coherent multi-copy Debye
phases---the multidimensional Weyl sum on which $F$ rests---is settled in \S\ref{sub:multicopy}
(Theorem~\ref{thm:joint-equi}) and its uniform-in-$u$ transfer makes $F$ the unconditional $t>N/2$ profile
(Theorem~\ref{thm:uniform-transfer}); what remains for $\limsup_N B_N/\sqrt N=\beta_{\mathrm{odd}}$ is only the
profile-maximizer inequality $\max_{s\ge1}F(s)=F(1)$ (Lemma~\ref{lem:varfloor} and its residual Pearson-walk
bound). That maximizer step aside, the difficulty is
intrinsic: Erd\H{o}s asked whether every unimodular polynomial must satisfy
$\max|P|\ge(1+c)\sqrt{n}$~\cite{Erdos57}; Kahane's ultraflat polynomials answered no~\cite{Kahane80}, with
the sharpest construction in~\cite{BombieriBourgain}, and the \emph{quadratic} chirp
(Gauss--Fresnel/Talbot) is even $L^1$-flat with ratio $\to1$. So $\limsup<1$ is an \emph{anti}-flatness
statement particular to the non-quadratic $\sin^2$ dispersion---precisely an Erd\H{o}s-type flatness
question for one structured family, and no generic argument supplies it. (A single-period
saddle van der Corput estimate fails: it predicts $0.61$ against the actual $0.93$, missing the Poisson-copy
coherence---which is precisely why the correct object is the \emph{joint} multi-copy equidistribution of
Theorem~\ref{thm:joint-equi}, not a single saddle. A fourth-moment/merit-factor route cannot supply the upper bound either:
$\max\{\|K\|_1/\sqrt N:\|K\|_2=1,\,N\|K\|_4^4=c\}=1-\Theta(\sqrt{c/N})$, a single spike of mass
$\sqrt{c/N}$ absorbing any fourth moment at negligible $\ell^1$ cost---so no bound below $1$ can follow
from $\|K\|_2$ and $\|K\|_4$ alone.)
\end{remark}

\begin{proposition}[two-copy profile and its maximizer]\label{prop:bn-profile}
For $t=sN/2$ with $s\in[0,1]$, the two-copy Debye approximation of Remark~\ref{rem:bn-parity} gives
$\|K_N(\cdot,t)\|_1=\bigl(F(s)+o(1)\bigr)\sqrt N$ with the \emph{explicit scalar profile}
\[
  F(s)=\begin{cases}
    c_0\sqrt s, & 0\le s\le\tfrac12,\\[3pt]
    \sqrt{\tfrac2\pi}\Bigl[\tfrac4\pi(1-s)\,A(s)+(2s-1)^{3/4}B(s)\Bigr], & \tfrac12\le s\le1,
  \end{cases}
\]
\[
  A(s)=\int_0^1\!\bigl(s^2-(1-s)^2w^2\bigr)^{-1/4}dw,\qquad
  B(s)=\int_0^1\! g\bigl(\widehat a,\widehat b\bigr)\,dw,
\]
where $\widehat a=[(1-w)(1+(2s-1)w)]^{-1/2}$, $\widehat b=[w(2s-(2s-1)w)]^{-1/2}$, and
$g(a,b)=\mathbb E_{\Phi,\Psi}\sqrt{a\cos^2\Phi+b\cos^2\Psi}$ with $\Phi,\Psi$ independent uniform.
Equivalently, in the rescaled site $u=k/N$,
\[
  F(s)=\sqrt{\tfrac2\pi}\int_0^1 g\bigl((s^2-u^2)_+^{-1/2},\,(s^2-(1-u)^2)_+^{-1/2}\bigr)\,du ,
\]
a single integral with $g(a,0)=\tfrac2\pi\sqrt a$, and the one-copy factor is the closed form
$A(s)=s^{1/2}(1-s)^{-1}\,x\,{}_2F_1\bigl(\tfrac14,\tfrac12;\tfrac32;x^2\bigr)$, $x=(1-s)/s$~\cite[\S15]{DLMF}.
Then $F(\tfrac12)=c_0/\sqrt2$ and $F(1)=\beta_{\mathrm{odd}}$ \eqref{eq:beta-odd}, and
\[
  \max_{s\in[0,1]}F(s)=F(1)=\beta_{\mathrm{odd}} .
\]
\end{proposition}

\begin{proof}[Proof]
The reduction and the closed-form endpoints are the Debye computation of Remark~\ref{rem:bn-parity}: for
$s\le\tfrac12$ the two Bessel fronts $|n|\approx sN$ are disjoint and each half-line contributes
$\tfrac12c_0\sqrt{2t}$, giving $c_0\sqrt s$; for $s\ge\tfrac12$ the overlap $u=k/N\in[1-s,s]$ carries both
Debye envelopes, which for odd $N$ add in quadrature, producing $A,B$ after $u\mapsto w$ maps the two
windows to $[0,1]$. The substitution makes the endpoint singularities $w^{-1/4},(1-w)^{-1/4}$ integrable
and the integrands smooth in $s$ on $(\tfrac12,1]$; at $s=1$ one recovers exactly \eqref{eq:beta-odd}.
For the maximizer, two analytic majorants of $g$ cover the bulk. Jensen's bound $g(a,b)\le\sqrt{(a+b)/2}$
($\mathbb E\cos^2=\tfrac12$) gives $F<\beta_{\mathrm{odd}}$ on $[\tfrac12,\tfrac45]$; writing
$g(a,b)=\sqrt{\max}\,h(\min/\max)$ with $h(\rho)=\mathbb E\sqrt{\cos^2\Phi+\rho\cos^2\Psi}$ \emph{concave}
($h''<0$ on $(0,1]$), the tangent at $\rho=1$, $h(\rho)\le h(1)+h'(1)(\rho-1)$ with $h(1)=0.95809$,
$h'(1)=0.23952$, sharpens this to $F<\beta_{\mathrm{odd}}$ on all of $[\tfrac12,0.937]$---an \emph{analytic}
bound on $87\%$ of the window, clearing the interior dip $\min F\approx0.847$ at $s\approx0.56$. On the
remaining $[0.937,1]$ the profile is strictly increasing, $F'(s)>0$, proved by \emph{validated interval
arithmetic}: two elementary inequalities eliminate the elliptic $g$---the QM--AM bound $g(a,b)\ge
\tfrac{\sqrt2}\pi(\sqrt a+\sqrt b)$ (from $\sqrt{x^2+y^2}\ge(x+y)/\sqrt2$) and the derivative bound
$\partial_ag=\mathbb E[\tfrac{\cos^2\Phi}{2\sqrt{\,\cdot\,}}]\le1/(\pi\sqrt a)$---and reduce $F'$ to a lower
bound $F_{\mathrm{lo}}(s)$ built from \emph{elementary} ($\sqrt{\ }$-only) integrals; an interval
rectangle-sum enclosure (the band-edge $w^{-1/4}$ singularities removed by $w=t^4$, $w=1-r^4$, powers
cancelled by hand), with $s$ itself taken as an interval, certifies $F_{\mathrm{lo}}(s)>0$ on every
subinterval of $[0.937,1]$ (worst case $0.051$, \texttt{verify\_bn\_profile\_rigorous.py}). Hence $F\le F(1)$
there, and the maximizer $\max_{[0,1]}F=F(1)=\beta_{\mathrm{odd}}$ is proved.
\end{proof}

\noindent This settles the $t\le N/2$ supremum: $\sup_{t\le N/2}\|K_N(\cdot,t)\|_1=(\beta_{\mathrm{odd}}
+o(1))\sqrt N$, since the maximization of the one explicit elliptic-integral function $F$ is now proved in
full---analytically on $[\tfrac12,0.937]$ and by validated interval arithmetic on $[0.937,1]$. The only
remaining piece of the constant is the multi-copy region $t>N/2$ (Remark~\ref{rem:bn-parity}).

\subsection{The multi-copy region \texorpdfstring{$t>N/2$}{t>N/2}: joint equidistribution of the Debye phases}\label{sub:multicopy}

Beyond $t=N/2$, three or more Poisson copies overlap and the two-copy reduction of
Proposition~\ref{prop:bn-profile} is no longer available. We isolate the exact analytic content of the
multi-copy regime, prove the one structural input that the heuristic profile rests on---the \emph{joint}
equidistribution of the overlapping Debye phases---and state honestly the single point that prevents this
from closing the $t>N/2$ bound unconditionally.

Fix a ratio $s=2t/N>1$ and the rescaled site $u=k/N\in[0,1)$. By the periodization \eqref{eq:poisson},
$K_N(k,t)=\sum_\ell \mathrm i^{\,k+\ell N}J_{k+\ell N}(2t)$, and a copy $\ell$ lies in the oscillatory band
exactly when its scaled order $m_\ell:=u+\ell$ satisfies $|m_\ell|<s$. There are at most $2s+2$ such
\emph{live} copies---a number bounded independently of $N$---and the band-edge copies $|m_\ell|=s+O(N^{-1})$
contribute negligibly (Airy front, height $O(N^{-1/3})$; the super-exponential tail handles $|m_\ell|>s$ as
in Lemma~\ref{lem:bessel-l1}). On the live set the uniform large-order asymptotics~\cite[\S10.19]{DLMF}
give, with the signed order $n_\ell=k+\ell N$ (so $J_{n_\ell}=(-1)^{|n_\ell|}J_{|n_\ell|}$ for $n_\ell<0$),
\begin{equation}\label{eq:multicopy-debye}
  \mathrm i^{\,n_\ell}J_{n_\ell}(2t)
  =\mathrm i^{\,n_\ell}\,\sqrt{\tfrac2{\pi N}}\,\bigl(s^2-m_\ell^2\bigr)^{-1/4}\,
   \bigl(\cos\Phi_\ell(k)+O(N^{-1})\bigr),
\end{equation}
with $\Phi_\ell(k)=\sqrt{x^2-n_\ell^2}-|n_\ell|\arccos(|n_\ell|/x)-\tfrac\pi4$,
$x=2t=sN$, uniformly on $u\in[\delta,s-\ell-\delta]\cap[\ell-s+\delta,\,\cdot\,]$ for each $\ell$ (we use the
phase of $|J_{n_\ell}|$; since $\cos$ is even its overall sign is immaterial, and the $(-1)^{|n_\ell|}$ of
$n_\ell<0$ is absorbed into $\mathrm i^{n_\ell}$). The phase $\Phi_\ell$ is the object that decides coherence;
its discrete derivative in the site index is exact and elementary.

\begin{lemma}[Debye phase increments and their curvature]\label{lem:debye-incr}
For each live copy $\ell$ and $x=sN$, with $g_\ell(u)=\bigl(s^2-m_\ell^2\bigr)^{-1/2}>0$, $m_\ell=u+\ell$,
\begin{align*}
  \Phi_\ell(k+1)-\Phi_\ell(k)&=-\operatorname{sgn}(m_\ell)\,\arccos\tfrac{|m_\ell|}{s}\pmod{2\pi},\\
  \Phi_\ell(k+1)-2\Phi_\ell(k)+\Phi_\ell(k-1)&=\tfrac1N\,g_\ell(u)+O(N^{-2}).
\end{align*}
Consequently, writing
$\theta_\ell(u)=-\operatorname{sgn}(m_\ell)\arccos(|m_\ell|/s)$ for the first increment,
$\theta_\ell'(u)=g_\ell(u)$.
\end{lemma}

\begin{proof}
$\Phi_\ell(k)=\phi(|n_\ell|)$ with $\phi(\nu)=\sqrt{x^2-\nu^2}-\nu\arccos(\nu/x)-\tfrac\pi4$ and
$\phi'(\nu)=-\arccos(\nu/x)$. Since $n_\ell=k+\ell N$, advancing $k\mapsto k+1$ advances $\nu=|n_\ell|$ by
$\operatorname{sgn}(n_\ell)$, so the first difference is $\operatorname{sgn}(n_\ell)\,\phi'(|n_\ell|)
=-\operatorname{sgn}(m_\ell)\arccos(|m_\ell|/s)$ (using $\operatorname{sgn}(n_\ell)=\operatorname{sgn}(m_\ell)$ and
$|n_\ell|/x=|m_\ell|/s$). Hence $\theta_\ell(u)=-\operatorname{sgn}(m_\ell)\arccos(|m_\ell|/s)$, and
differentiating, $\theta_\ell'(u)=\operatorname{sgn}(m_\ell)\cdot\dfrac{\operatorname{sgn}(m_\ell)}{\sqrt{s^2-m_\ell^2}}
=(s^2-m_\ell^2)^{-1/2}=g_\ell(u)>0$; the second site-difference is $N^{-1}\theta_\ell'(u)+O(N^{-2})$ by Taylor
expansion. The increment identities are verified to machine precision (\texttt{verify\_debye\_equidistribution.py}:
ratio $1.0000$ for the second difference against $N^{-1}g_\ell$).
\end{proof}

The first increment $\theta_\ell(u)$ varies with $u$, so each single phase $\{\Phi_\ell(k)\}_k$
equidistributes mod $2\pi$ (the classical van der Corput statement behind Lemma~\ref{lem:bessel-l1}). The new
content is that the live phases equidistribute \emph{jointly}.

\begin{theorem}[joint equidistribution of the live Debye phases]\label{thm:joint-equi}
Fix $s>1$ and a cell $\mathcal C=(a,b)\subset[0,1)$ on which the live set $L(s)=\{\ell:|u+\ell|<s\}$ is
constant and no $m_\ell$ meets the band edge. As $N\to\infty$ the vector
$\bigl(\Phi_\ell(k)\bmod2\pi\bigr)_{\ell\in L(s)}$ is equidistributed in the torus $\mathbb T^{|L(s)|}$ as $k$
ranges over $\{k:k/N\in\mathcal C\}$. Quantitatively, for every nonzero integer vector $(k_\ell)_{\ell\in L(s)}$,
\begin{equation}\label{eq:joint-weyl}
  \Bigl|\frac1{\#}\sum_{k/N\in\mathcal C}\eu^{\mathrm i\sum_\ell k_\ell\Phi_\ell(k)}\Bigr|
  \ \ll_{s,\mathcal C,k}\ N^{-1/2},
\end{equation}
the power saving of the second-derivative van der Corput estimate; hence by Weyl's criterion the joint
distribution is uniform, with Erd\H{o}s--Tur\'an discrepancy $O(N^{-1/2}\log N)$ on the cell.
\end{theorem}

\begin{proof}
By Weyl's criterion~\cite{Weyl}\cite[Ch.~1]{KN} joint equidistribution is equivalent to \eqref{eq:joint-weyl}
for each nonzero $(k_\ell)$. Set $\Psi_k(j):=\sum_{\ell\in L(s)}k_\ell\Phi_\ell(j)$. By Lemma~\ref{lem:debye-incr} its
second difference is
\[
  \Psi_k(j+1)-2\Psi_k(j)+\Psi_k(j-1)=\frac1N\,C_k(u)+O(N^{-2}),\qquad
  C_k(u)=\sum_{\ell\in L(s)}\frac{k_\ell}{\sqrt{s^2-(u+\ell)^2}} ,
\]
so, viewing $\Psi_k$ as a smooth phase $f(j)$ with $f''(j)=N^{-1}C_k(j/N)+O(N^{-2})$, the van der Corput
second-derivative test (the exponent-pair $(\tfrac12,\tfrac12)$ bound, \cite[\S8]{Montgomery},
\cite[Ch.~2]{GrahamKolesnik}) gives, on a block of $L=\Theta(N)$ consecutive sites with
$|f''|\asymp\lambda=N^{-1}|C_k|$,
\[
  \Bigl|\sum_{j} \eu^{\mathrm i f(j)}\Bigr|\ \ll\ L\,\lambda^{1/2}+\lambda^{-1/2}
   \ \ll\ N\cdot N^{-1/2}|C_k|^{1/2}+N^{1/2}|C_k|^{-1/2}\ \ll\ N^{1/2},
\]
which after dividing by $\#=\Theta(N)$ is \eqref{eq:joint-weyl}. The estimate requires $C_k$ bounded away
from $0$; here is the only structural point. The functions $u\mapsto g_\ell(u)=(s^2-(u+\ell)^2)^{-1/2}$,
$\ell\in L(s)$, have \emph{distinct} singularities at $u=\pm s-\ell$, so they are linearly independent over
$\mathbb R$: any relation $\sum_\ell k_\ell g_\ell\equiv0$ on an interval extends, by real-analyticity, to a
relation between functions with distinct poles, forcing every $k_\ell=0$. Hence $C_k\not\equiv0$; being
real-analytic on $\mathcal C$ it has only finitely many zeros there, and on the complement of
$\varepsilon$-neighbourhoods of those zeros $|C_k|\ge c(s,\mathcal C,k,\varepsilon)>0$ and the bound above
applies. Each exceptional $\varepsilon$-neighbourhood has length $O(\varepsilon)$ and contributes
$\le\varepsilon$ to the normalized sum trivially; letting $\varepsilon\to0$ after $N\to\infty$ removes it. The
discrepancy bound is the Erd\H{o}s--Tur\'an inequality~\cite[Ch.~2]{KN} applied to \eqref{eq:joint-weyl}. The Weyl sums
are computed directly: \eqref{eq:joint-weyl} decays with fitted exponent $-0.52$ to $-1.2$ across nonzero
$(k_\ell)$ with $|k_\ell|\le3$ (the slowest matching the predicted $N^{-1/2}$), the worst over $60$ random
$(k_\ell)$ at $N=2^{18}$ being $0.013$ (\texttt{verify\_debye\_equidistribution.py}).
\end{proof}

\noindent Theorem~\ref{thm:joint-equi} is the rigorous core the heuristic profile of
\texttt{verify\_bn\_residual\_profile.py} assumes. It upgrades the $s\le1$ two-copy statement
(Proposition~\ref{prop:bn-profile}, where $|L(s)|=2$) to all $s$. The cell-wise statement is not yet the
$L^1$ \emph{limit}: \eqref{eq:joint-weyl} carries an implied constant depending on the cell, and the limit
needs the discrepancy $o(1)$ \emph{uniformly} in $u$, including the finitely many band edges and coalescing
points where the constant degenerates. We supply that uniform control now; it is what turns
Theorem~\ref{thm:joint-equi} into the unconditional profile of Proposition~\ref{prop:multicopy}.

\begin{theorem}[uniform-in-$u$ transfer]\label{thm:uniform-transfer}
Fix $s>1$. With the rescaled kernel of \eqref{eq:multicopy-debye} and the toral integrand
$\mathcal E(u):=\mathbb E_{(\Phi_\ell)\sim U}\bigl|\sum_{\ell\in L(s)}c_\ell(s^2-(u+\ell)^2)^{-1/4}\cos\Phi_\ell\bigr|$,
\[
  \frac1{\sqrt N}\,\bigl\|K_N(\cdot,sN/2)\bigr\|_1\ \xrightarrow[N\to\infty]{}\ \sqrt{\tfrac2\pi}\int_0^1\mathcal E(u)\,du\ =\ F(s).
\]
The convergence is at an explicit polynomial rate $O(N^{-\eta})$, $\eta=\tfrac1{9|L(s)|}>0$ (the cubic
exponent pair $\tfrac19$ at the coalescing points, divided by $|L|$ for the Lipschitz-cone mollification of
the modulus in Koksma--Hlawka; numerically the cores decay like $N^{-1}$, far faster, so the true rate is
band-edge-strip-limited at $N^{-1/4+\delta'}$).
\end{theorem}

\begin{proof}
Write $W_N(k):=\sqrt{\pi N/2}\,|K_N(k,sN/2)|$, so that $\|K_N\|_1/\sqrt N=\sqrt{2/(\pi N)}\,\sum_k|K_N(k)|
=\sqrt{2/\pi}\cdot N^{-1}\sum_{k=0}^{N-1}W_N(k)$, the normalized prefactor being $\sqrt{2/\pi}$ (the order
density has exactly $N$ sites per unit of $u=k/N$, and the $s$-dependence sits inside $W_N$ through the
envelope; numerically $\|K_N\|_1/\sqrt N$ tracks $\sqrt{2/\pi}\int_0^1\mathcal E\,du$ across $s\in\{1.5,2.7,4\}$,
while $\sqrt{2/(\pi s)}$ is off by $\sqrt s$, \texttt{verify\_bandedge\_transfer.py}). It suffices to show
\begin{equation}\label{eq:transfer-target}
  \frac1N\sum_{k=0}^{N-1}W_N(k)\ \longrightarrow\ \int_0^1\mathcal E(u)\,du .
\end{equation}
Let $E=\{u\in[0,1):|u+\ell|=s\text{ for some }\ell\}$ be the (finite) set of band edges and
$Z=\{u^*=-(\ell+\ell')/2:\ell\ne\ell'\in L,\ u^*\in[0,1)\}$ the (finite) set of coalescing points, where a
\emph{pair} $g_\ell,g_{\ell'}$ collides; these are geometric, fixed points of $[0,1)$, independent of the
test modes $k$. Fix a small constant $\rho>0$ (independent of $N$, chosen below) and a small exponent
$0<\delta<\tfrac16$, and put $\varepsilon=N^{-2/3+\delta}$. Partition $[0,1)$ into
\[
  \mathcal S=\!\!\bigcup_{u_0\in E}\!(u_0-\varepsilon,u_0+\varepsilon),\qquad
  \mathcal K=\!\!\bigcup_{u^*\in Z}\!(u^*-\rho,u^*+\rho)\setminus\mathcal S,\qquad
  \mathcal B=[0,1)\setminus(\mathcal S\cup\mathcal K),
\]
the band-edge strips (width $\varepsilon\to0$), the coalescing cores (fixed width $\rho$), and the bulk. We
bound the contribution of each to \eqref{eq:transfer-target}.

\emph{(a) Bulk $\mathcal B$.} By construction every $u\in\mathcal B$ is at distance $\ge\rho$ from $Z$ and
$\ge\varepsilon$ from $E$. On each maximal subinterval $\mathcal C\subset\mathcal B$ the live set $L$ is
constant, the envelopes $a_\ell(u)=(s^2-(u+\ell)^2)^{-1/4}$ are bounded and $C^1$, so the rescaled kernel is a
fixed function $W_N(k)=G_u(\Phi(k))$ of the phase vector, $G_u\colon\mathbb T^{|L|}\to\mathbb R$ the modulus of
a trigonometric polynomial with $u$-uniformly bounded, $C^1$ coefficients, $\widehat G_u(0)=\mathbb
E_{(\Phi_\ell)\sim U}G_u=\mathcal E(u)$. We do \emph{not} Fourier-expand $G_u$ (a modulus is only Lipschitz, its
coefficients too slow to sum against the van der Corput bound); instead we bound the \emph{star-discrepancy}
$D^*_N(\mathcal C)$ of the finite phase-point set $\{(\Phi_\ell(k))_{\ell\in L}:k/N\in\mathcal C\}$ in
$\mathbb T^{|L|}$ and invoke Koksma--Hlawka. A modulus is only Lipschitz, and for $|L|\ge3$ its Hardy--Krause
variation is in fact \emph{infinite}: the cone point at $Z=0$ (codimension $2$) makes the top mixed partial
grow like $|Z|^{-(|L|-1)}$, and $\int|Z|^{-(|L|-1)}\,dX\,dY\asymp\int r^{-(|L|-2)}\,dr$ diverges
(logarithmically already at $|L|=3$; \texttt{verify\_bandedge\_transfer.py} confirms the $\log N$ growth of the
Vitali variation on a refining grid). So Koksma--Hlawka cannot be applied to $G_u$ itself. Instead we use only
that $G_u$ is \emph{Lipschitz}, $\operatorname{Lip}(G_u)\le\sum_\ell a_\ell\le\Lambda(s)$, and mollify at scale
$\eta$: $G_u^\eta=G_u*\varphi_\eta$ obeys $\|G_u-G_u^\eta\|_\infty\le\Lambda\eta$ and the smooth bound
$V(G_u^\eta)\le C(s)\,\eta^{-(|L|-1)}$. Applying Koksma--Hlawka \cite[Ch.~2, Thm.~5.5]{KN} to $G_u^\eta$ and
absorbing the two mollification errors,
\begin{equation}\label{eq:koksma}
  \Bigl|\frac1{\#}\sum_{k/N\in\mathcal C}G_u(\Phi(k))-\mathcal E(u)\Bigr|\ \le\
  2\Lambda(s)\,\eta+C(s)\,\eta^{-(|L|-1)}D^*_N(\mathcal C);
\end{equation}
optimizing $\eta=D^*_N(\mathcal C)^{1/|L|}$ yields the Lipschitz form
$\bigl|\overline{G_u}-\mathcal E(u)\bigr|\le C'(s)\,D^*_N(\mathcal C)^{1/|L|}$, valid despite the unbounded
variation (the $1/|L|$ power is the price of mollifying a cone).
The discrepancy is controlled by the Erd\H{o}s--Tur\'an--Koksma inequality \cite[Ch.~2, Thm.~5.6]{KN}:
\[
  D^*_N(\mathcal C)\ \ll_{|L|}\ \frac1M+\sum_{0<\|k\|_\infty\le M}\frac1{r(k)}\,
   \Bigl|\frac1{\#}\sum_{k'/N\in\mathcal C}\eu^{\mathrm i\langle k,\Phi(k')\rangle}\Bigr|,
   \qquad r(k)=\prod_\ell\max(1,|k_\ell|),
\]
the weight $1/r(k)$ (not the Fourier coefficients of $G_u$) summing to $\sum_{0<\|k\|_\infty\le M}1/r(k)
=O(\log^{|L|}M)$. Each Weyl sum is the \emph{integrated} second-derivative van der Corput estimate, uniform in
$k$ by two structural facts. First, the number of zeros of $C_k=\sum_\ell k_\ell g_\ell$ on $\mathcal C$ is
bounded \emph{independently of $k$}: clearing the $|L|$ square roots in
$C_k=\sum_\ell k_\ell(s^2-(u+\ell)^2)^{-1/2}$ over a common factor leaves an algebraic function whose zero
count is $\le2(|L|-1)$, a degree fixed by the live set, not by $k$ (verified: $\le1$ zero over all
$\|k\|_\infty\le6$ on a three-copy cell). Second, every such zero \emph{in the bulk} is \emph{simple}: a double
zero $C_k(u_0)=C_k'(u_0)=0$ would need the integer vector $k$ orthogonal to both $g(u_0)=(g_\ell(u_0))_\ell$ and
$g'(u_0)$, vectors linearly independent off the discrete coalescing set $Z$; so $|C_k'(u_0)|\gtrsim\|k\|>0$ at
every bulk zero (checked: $\min|C_k'|/\|k\|=0.15$ over $\|k\|_\infty\le6$). Hence on $\mathcal C$ each $C_k$ has
$\le2(|L|-1)$ simple zeros and $\int_{\mathcal C}|C_k(u)|^{-1/2}\,du\le A(s)<\infty$ \emph{uniformly in $k$} (the
$|u-u_0|^{-1/2}$ singularity is integrable). Excising the $N^{-1/3}$-neighbourhoods of those zeros and applying
Lemma~\ref{lem:vdc-explicit} on each remaining run, with $|f''|=N^{-1}|C_k|+O(N^{-2})$, then Riemann-summing
the bound $\#\lambda^{1/2}+\lambda^{-1/2}$ in $u$ (the $\lambda^{-1/2}=N^{1/2}|C_k|^{-1/2}$ term against
$\int|C_k|^{-1/2}du\le A(s)$),
\[
  \Bigl|\frac1{\#}\sum_{k'/N\in\mathcal C}\eu^{\mathrm i\langle k,\Phi(k')\rangle}\Bigr|
  \ \ll_s\ N^{-1/2}A(s)\,\|k\|^{1/2}\ +\ N^{-1/3} .
\]
Feeding this into Erd\H{o}s--Tur\'an--Koksma, and using $\sum_{0<\|k\|_\infty\le M}\|k\|^{1/2}/r(k)
\le M^{1/2}\sum 1/r(k)=O(M^{1/2}\log^{|L|}M)$,
\[
  D^*_N(\mathcal C)\ \ll_s\ \frac1M+N^{-1/2}M^{1/2}\log^{|L|}\!M+N^{-1/3}\log^{|L|}\!M ,
\]
which at $M=N^{1/3}$ is $D^*_N(\mathcal C)=O_s(N^{-1/3}\log^{|L|}N)$. By the Lipschitz form of
\eqref{eq:koksma} the cell average then converges to $\mathcal E(u)$ at the mollified rate
$D^*_N{}^{1/|L|}=O_s(N^{-1/(3|L|)}\log N)$, and a $u$-Riemann sum over the $\Theta(N^{1/3})$ panels of
$\mathcal C$ (the integrand $\mathcal E$ being $C^1$ on $\mathcal B$) gives
\[
  \frac1{N}\sum_{k/N\in\mathcal B}W_N(k)\ =\ \int_{\mathcal B}\mathcal E(u)\,du\ +\ O_s(N^{-1/(3|L|)}\log N),
\]
uniformly. This is the uniform replacement for the cell-wise limit \eqref{eq:joint-weyl}: the only inputs are
the $k$-uniform integral $A(s)$, the Lipschitz constant $\Lambda(s)$, the $k$-uniform zero count and the
simplicity of bulk zeros (so $|C_k|^{-1/2}$ is integrable).

\emph{(b) Band-edge strips $\mathcal S$.} On a strip $(u_0-\varepsilon,u_0+\varepsilon)$ around an edge
$u_0=s-\ell$ (a copy entering/leaving), the entering envelope is $a_\ell(u)=(s^2-(u+\ell)^2)^{-1/4}=(2s)^{-1/4}
(u_0-u)^{-1/4}(1+O(\varepsilon))$ for $u<u_0$; the other live envelopes stay bounded. By the triangle
inequality $W_N(k)\le\sum_{\ell\in L}a_\ell(k/N)$, so
\[
  \frac1N\sum_{k/N\in\mathcal S}W_N(k)\ \le\ \sum_{\ell\in L}\int_{\mathcal S}a_\ell(u)\,du+O(N^{-1})
  \ =\ O\!\Bigl(\!\int_0^{\varepsilon}\!(2s)^{-1/4}r^{-1/4}\,dr\Bigr)
  \ =\ O\bigl(\varepsilon^{3/4}\bigr),
\]
the integrable envelope singularity giving the exponent $\tfrac34$ (the leading constant is
$\tfrac43(2s)^{-1/4}$ per edge; \texttt{verify\_bandedge\_transfer.py} confirms the law and the constant to
$0.4\%$). With $\varepsilon=N^{-2/3+\delta}$ this is $O(N^{-1/2+3\delta/4})\to0$. The same bound caps the
\emph{toral} side: $\int_{\mathcal S}\mathcal E(u)\,du\le\sum_\ell\int_{\mathcal S}a_\ell=O(\varepsilon^{3/4})$,
since $\mathcal E(u)\le\mathbb E|\sum_\ell\!\pm a_\ell\cos\Phi_\ell|\le\sum_\ell a_\ell$. So both the empirical
and the toral mass of the strips are $O(\varepsilon^{3/4})$ and cancel in \eqref{eq:transfer-target} up to that
order. The width is chosen so that the \emph{retained} cells adjacent to the edge still wind: at distance
$\varepsilon$ from $u_0$ the single-copy increment is
$\theta_\ell=\arccos(|m_\ell|/s)=\arccos(1-\varepsilon/s+\cdots)\asymp(2\varepsilon/s)^{1/2}$, so the live phase
$\Phi_\ell$ advances by $\asymp\varepsilon^{1/2}$ per site over $\asymp\varepsilon N$ sites, a total winding
$\asymp\varepsilon^{3/2}N=N^{3\delta/2}\to\infty$; thus part (a) applies on $\mathcal B\cap\{$near edge$\}$
without modification (the curvature floor there is in fact \emph{larger}, $g_\ell\asymp\varepsilon^{-1/2}$). The
choice $\varepsilon=N^{-2/3+\delta}$ with $0<\delta<\tfrac16$ keeps both: strip mass $o(1)$ and retained
winding $\to\infty$.

\emph{(c) Coalescing cores $\mathcal K$.} At $u^*=-(\ell+\ell')/2$ one has $(u^*+\ell)^2=(u^*+\ell')^2$, so
$g_\ell(u^*)=g_{\ell'}(u^*)$ and the difference combination $C_{(1,-1)}=g_\ell-g_{\ell'}$ vanishes: the
second-derivative test \eqref{eq:joint-weyl} degenerates. The third derivative does not. Differentiating
$g_\ell(u)=(s^2-(u+\ell)^2)^{-1/2}$ gives $g_\ell'(u)=(u+\ell)(s^2-(u+\ell)^2)^{-3/2}$, and at $u^*$, where
$u^*+\ell=\tfrac{\ell-\ell'}2$ and $u^*+\ell'=-\tfrac{\ell-\ell'}2$ have \emph{equal squares} but opposite
signs,
\[
  \bigl(g_\ell-g_{\ell'}\bigr)'(u^*)
  =\frac{(\ell-\ell')/2-(-(\ell-\ell')/2)}{\bigl(s^2-(\tfrac{\ell-\ell'}2)^2\bigr)^{3/2}}
  =\frac{\ell-\ell'}{\bigl(s^2-(\tfrac{\ell-\ell'}2)^2\bigr)^{3/2}}\ \ne0 ,
\]
a clean nonzero value (verified to machine precision, \texttt{verify\_bandedge\_transfer.py}). For a general
$k$ with $C_k(u^*)=0$, $C_k$ is real-analytic and $\not\equiv0$ (distinct poles, Theorem~\ref{thm:joint-equi}),
so its zero at $u^*$ is isolated and of finite order. We show it is simple. Two of the $g_\ell$ can coincide
at a point only pairwise: $g_\ell=g_{\ell'}\Leftrightarrow u=-(\ell+\ell')/2$, and three pairwise-equal
averages $-(\ell+\ell')/2=-(\ell+\ell'')/2$ force $\ell'=\ell''$, so \emph{no triple coincidence} occurs
(checked over a $10^5$-grid: $0$ hits). Shrink $\rho$ once and for all so that each $(u^*-\rho,u^*+\rho)$
contains exactly the one pair colliding at $u^*$ and no other point of $Z$; then on the punctured core
$C_k(u)=C_k'(u^*)(u-u^*)+O((u-u^*)^2)$ with the transverse slope
$C_k'(u^*)=k_\ell\,g_\ell'(u^*)+k_{\ell'}g_{\ell'}'(u^*)$. If $C_k'(u^*)\ne0$ the zero is simple; the residual
case $C_k'(u^*)=0$ forces $k_\ell=k_{\ell'}$ (the two slopes are equal and opposite by the displayed identity),
i.e.\ the colliding modes enter with the \emph{same} weight, which is exactly the $C_{(1,-1)}$-type difference
whose nonzero third derivative is displayed above---so $C_k$ then has a simple \emph{cubic} inflection rather
than a simple zero. In both sub-cases $C_k$ grows at least linearly off a vanishing inner core; we use only
that. Split $\mathcal K$ into the inner cores $\mathcal K_0=\bigcup_{u^*}(u^*-h,u^*+h)$, $h=N^{-1/3}$, and the
shoulder $\mathcal K_1=\mathcal K\setminus\mathcal K_0$.

We transfer $\mathcal K_1$ by the same Koksma--Hlawka/Erd\H{o}s--Tur\'an--Koksma route as (a). For the
simple-zero modes ($C_k'(u^*)\ne0$, the generic $k$) $|C_k(u)|\asymp|C_k'(u^*)|\,|u-u^*|$ has the integrable
$|u-u^*|^{-1/2}$ singularity and the second-derivative bound $N^{-1/2}A(s)\|k\|^{1/2}+N^{-1/3}$ of (a) holds
unchanged. The \emph{difference} modes ($k_\ell=k_{\ell'}$ on the colliding pair) instead have the cubic
inflection $|C_k(u)|\asymp\kappa|u-u^*|^2$, $\kappa=|(g_\ell-g_{\ell'})'(u^*)|>0$. Here the curvature is
governed by the nonvanishing \emph{third} derivative, and the natural tool is the cubic ($r=3$) van der
Corput / exponent-pair $(\tfrac16,\tfrac46)$ bound \cite[\S2.3]{GrahamKolesnik}: with $|f'''|\asymp N^{-2}\kappa$
on a run of length $L$, $|\sum_j\eu^{\mathrm if(j)}|\ll L(N^{-2}\kappa)^{1/6}+(N^{-2}\kappa)^{-1/3}$, whose
normalized value $O(N^{-1/9})$ feeds Erd\H{o}s--Tur\'an--Koksma with the $1/r(k)$ weight (the difference
directions being finite in number, their integer multiples summable against $1/r$). (Equivalently, because
$\mathcal K_1$ already excludes the inner core, even the cubic-inflection density is \emph{barely} integrable
there, $\int_{\mathcal K_1}|C_k|^{-1/2}du\asymp\kappa^{-1/2}\log(\rho/h)=O(\log N)$, so the second-derivative
bound of (a) also closes the shoulder with one extra $\log$---the cubic test is the sharper of the two and is
the structurally correct object at the degenerate point.) The resulting shoulder discrepancy is
$O_s(N^{-1/9}\log^{|L|}N)$. Finally the inner cores $\mathcal K_0=\bigcup_{u^*}(u^*-h,u^*+h)$, $h=N^{-1/3}$,
are discarded trivially: $W_N\le\sum_\ell a_\ell=O(1)$ there, so the empirical mass is
$\le\sum_\ell\int_{\mathcal K_0}a_\ell=O(h)=O(N^{-1/3})$ and the same bound caps the toral mass
$\int_{\mathcal K_0}\mathcal E\le O(h)$; the two cancel up to $O(N^{-1/3})$. (The width $h\asymp N^{-1/3}$ is the
optimum of the trivial inner bound $2h$ against the shoulder cost $N^{-1/2}\kappa^{-1/2}h^{-1/2}$, combined
cost $O(N^{-1/3})$, exponent confirmed in \texttt{verify\_bandedge\_transfer.py}.) Hence
\[
  \frac1N\sum_{k/N\in\mathcal K}W_N(k)\ =\ \int_{\mathcal K}\mathcal E(u)\,du\ +\ O_s(N^{-1/(9|L|)}\log N),
\]
(Numerically a full cell \emph{containing} an interior coalescing point already has its normalized Weyl sum
decay like $N^{-1}$, faster than $N^{-1/2}$, because the degenerate set has $u$-measure $O(N^{-1/3})$ and the
shoulder floor dominates: \texttt{verify\_bandedge\_transfer.py}, slope $-1.03$.)

\emph{Assembly.} Summing (a)--(c) over the $O(s)$ live-set cells,
\[
  \frac1N\sum_{k=0}^{N-1}W_N(k)=\int_0^1\mathcal E(u)\,du
   +\underbrace{O(N^{-1/(3|L|)}\log N)}_{\text{bulk}}+\underbrace{O(N^{-1/2+3\delta/4})}_{\text{strips}}
   +\underbrace{O(N^{-1/(9|L|)}\log N)}_{\text{cubic cores}},
\]
the band-edge mass $O(\varepsilon^{3/4})$ on the strips being removed from \emph{both} the empirical and the
toral sides (part (b)), so only its $O(\varepsilon^{3/4})=O(N^{-1/2+3\delta/4})$ residue survives. With
$0<\delta<\tfrac16$ every error is a positive power of $1/N$, hence $o(1)$; the binding rate is the mollified
cubic core $N^{-1/(9|L|)}\log N$---the $1/|L|$ being the Lipschitz-cone mollification price of
\eqref{eq:koksma}, the cubic $\tfrac19$ the off-the-shelf $(\tfrac16,\tfrac46)$ exponent pair. Multiplying by
$\sqrt{2/\pi}$ gives \eqref{eq:transfer-target} and the theorem, with any $\eta<\tfrac1{9|L(s)|}$. The
band-edge strip is the only \emph{structurally} unavoidable loss; the polynomial rate, though far from sharp
(numerically the cores decay like $N^{-1}$, slope $-1.03$), is a genuine positive power---all the $o(1)$
transfer needs. The assembled discrepancy is checked to vanish: the rescaled cell-averaged $|K_N|$ tracks
$\mathcal E$ to under $1\%$ with
the residual oscillating in sign as $N$ grows from $4\cdot10^3$ to $10^6$, strips excised
(\texttt{verify\_bandedge\_transfer.py}).
\end{proof}

\begin{lemma}[explicit second-derivative van der Corput / Erd\H{o}s--Tur\'an]\label{lem:vdc-explicit}
Let $f\in C^2[A,B]$ with $\lambda\le|f''|\le h\lambda$ ($h\ge1$, $f''$ of one sign) on $[A,B]$. Then
$\bigl|\sum_{A\le j\le B}\eu^{2\pi\mathrm i f(j)}\bigr|\le c_1\bigl((B-A)\lambda^{1/2}+\lambda^{-1/2}\bigr)$ with
an absolute $c_1$ \cite[Thm.~8.20]{Montgomery},\cite[Ch.~2]{GrahamKolesnik}. Consequently, for a sequence
$x_j\in\mathbb T^d$ whose every nonzero Fourier coefficient $\langle k,x_j\rangle$ is such a phase, the
Erd\H{o}s--Tur\'an inequality \cite[Ch.~2, Thm.~2.5]{KN} bounds the discrepancy by
$D_{[A,B]}\ll \tfrac1M+\sum_{1\le|k|\le M}\tfrac1{|k|}\bigl|\tfrac1{B-A}\sum_j\eu^{2\pi\mathrm i\langle k,x_j\rangle}\bigr|$,
and optimizing $M$ over the van der Corput bound gives $D=O(N^{-1/2}\log N)$ on a $\Theta(N)$-block with
$\lambda\asymp N^{-1}$, the rate quoted in Theorem~\ref{thm:joint-equi}.
\end{lemma}

\noindent (Lemma~\ref{lem:vdc-explicit} is the explicit constant behind the cell-wise
\eqref{eq:joint-weyl}; in the proof of Theorem~\ref{thm:uniform-transfer} it is applied per Fourier mode
$k$ with the $u$-dependent floor $\lambda(u)=N^{-1}|C_k(u)|$, the bound integrated in $u$ against the
$k$-uniform $\int|C_k|^{-1/2}du\le A(s)$ rather than against a pointwise floor---the device that makes the
transfer uniform across the high modes whose curvature vanishes inside the cell.)

\begin{proposition}[multi-copy profile]\label{prop:multicopy}
Let $s>1$ and let $L(s)$ denote the live set on the cell containing $u$. Define the toral profile
\begin{equation}\label{eq:Fmulti}
  F(s)=\sqrt{\tfrac2{\pi}}\int_0^1
   \mathbb E_{(\Phi_\ell)\stackrel{\mathrm{iid}}\sim U(0,2\pi)}
   \Bigl|\sum_{\ell\in L(s)}c_\ell\,\bigl(s^2-(u+\ell)^2\bigr)^{-1/4}\cos\Phi_\ell\Bigr|\,du,\qquad
   c_\ell=\mathrm i^{\,\ell N},
\end{equation}
(only $\mathrm i^{N}=\pm1$ (even $N$) or $\pm\mathrm i$ (odd $N$) enters, so $F$ has two parity branches).
Then $\|K_N(\cdot,sN/2)\|_1=(F(s)+o(1))\sqrt N$. Numerically $F$ matches the exact FFT $\ell^1$ norm to $0.5\%$
across all four residues $N\bmod4$ and all $s\in(0,2]$, equals $\beta_{\mathrm{odd}}$ at $s=1$ and
$c_0/\sqrt2$ at $s=\tfrac12$, and satisfies $F(s)<F(1)=\beta_{\mathrm{odd}}$ for every $s\ne1$ (odd branch)
and $F(s)<F(\tfrac12)=c_0/\sqrt2$ for $s\ne\tfrac12$ (even branch), stably out to $s=10$
(\texttt{verify\_bn\_residual\_profile.py}).
\end{proposition}

\begin{proof}
By \eqref{eq:multicopy-debye} the rescaled kernel on a cell is, up to $O(N^{-1})$,
$|K_N(k,t)|\sqrt{\tfrac{\pi N}2}=\bigl|\sum_{\ell\in L(s)}c_\ell\,(s^2-m_\ell^2)^{-1/4}\cos\Phi_\ell(k)\bigr|
=G_u(\Phi(k))$, a fixed continuous function $G_u\colon\mathbb T^{|L(s)|}\to\mathbb R$ of the phase vector (the
$j$-independent global factor $\mathrm i^{k}$ drops under $|\cdot|$, and $c_\ell=\mathrm i^{\ell N}$ carries the
only $N$-dependence), whose toral mean is $\widehat G_u(0)=\mathbb E_{(\Phi_\ell)\sim U}G_u(\Phi)=\mathcal E(u)$.
Theorem~\ref{thm:uniform-transfer} is exactly the statement $\|K_N(\cdot,sN/2)\|_1/\sqrt N\to\sqrt{2/\pi}
\int_0^1\mathcal E(u)\,du=F(s)$, the uniform-in-$u$ transfer of the cell-wise joint equidistribution
(Theorem~\ref{thm:joint-equi}) to the $L^1$ limit. The endpoint identities $F(\tfrac12)=c_0/\sqrt2$,
$F(1)=\beta_{\mathrm{odd}}$ are \eqref{eq:beta-odd} and Proposition~\ref{prop:bn-profile} read at $|L(s)|=2$;
the prefactor is $\sqrt{2/\pi}$ (one site per $du/N$), consistent with Lemma~\ref{lem:varfloor} and with the
exact FFT to $0.5\%$ at $s\in\{1.5,2.7,4\}$ (\texttt{verify\_bandedge\_transfer.py}; the $\sqrt{2/(\pi s)}$ of
earlier drafts overcounts by $\sqrt s$ and is corrected here).
\end{proof}

Granting the transfer, the profile \emph{maximizer} is itself controlled rigorously by an elementary
variance bound.

\begin{lemma}[variance floor; the profile does not beat the $s=1$ peak]\label{lem:varfloor}
Write $a_\ell=(s^2-(u+\ell)^2)^{-1/4}$ and split the live copies of each parity channel into a $K$-copy
\emph{core} and a remainder. With $c^2=\tfrac12\sum_{\ell\in\mathrm{rest}}a_\ell^2$,
\[
  G(u,s)=\mathbb E\sqrt{X^2+Y^2}\ \le\ F_K(u,s):=\mathbb E_{\mathrm{core}}\sqrt{X_C^2+Y_C^2+c^2},
\]
since conditioning on the core, the mean-zero remainder gives $\mathbb E_{\mathrm{rest}}(X^2+Y^2)=
X_C^2+Y_C^2+c^2$ (cross terms vanish) and $\sqrt{\,\cdot\,}$ is concave. The bound is exact at $K=|L(s)|$,
monotone in $K$, and $F_K(s):=\sqrt{2/\pi}\int_0^1F_K\,du\ge F(s)$. As $s\to\infty$,
$\sum_\ell a_\ell^2\to\int_{-s}^s(s^2-x^2)^{-1/2}\,dx=\pi$, the channels become Gaussian $\mathcal N(0,\pi/4)$,
and $F(s)\to\sqrt\pi/2=0.8862<\beta_{\mathrm{odd}}$, with margin $0.042$.
\end{lemma}

\noindent The variance-floor bound clears the peak on the compact window: $F_K(s)<\beta_{\mathrm{odd}}$ for
$s\in(1,6]$ (the binding sliver just above $s=1$, two-copy, has head-room $0.007$; the bulk $0.03$),
validated by seeded Monte-Carlo with a $3\sigma$ upper-confidence bound and a deterministic FFT cross-check,
and the tail extends it to all $s$ (\texttt{verify\_bn\_largewave\_tail.py}). A cleaner route isolates the
gap as a single \emph{Gaussian-comparison} inequality. Let $\sigma_X^2=\tfrac12\sum_{\ell\ \mathrm{even}}a_\ell^2$,
$\sigma_Y^2=\tfrac12\sum_{\ell\ \mathrm{odd}}a_\ell^2$, $\Sigma_2=\sigma_X^2+\sigma_Y^2$, and let
$G=(G_X,G_Y)\sim\mathcal N(0,\sigma_X^2)\times\mathcal N(0,\sigma_Y^2)$ be the matched Gaussian. The expected
modulus of $G$ is the elementary closed form $\mathbb E|G|=\sigma_{\max}\sqrt{2/\pi}\,E\bigl(1-\sigma_{\min}^2/\sigma_{\max}^2\bigr)$
($E$ the complete elliptic integral of the second kind; at $s=1$, $\sigma_X=\sigma_Y$, this is $\sqrt\pi/2$),
so the pure-Gaussian profile $F_{\mathrm{gauss}}(s)=\sqrt{2/\pi}\int_0^1\mathbb E|G|\,du$ lies in $0.835$--$0.883$,
\emph{strictly below} $\beta_{\mathrm{odd}}$ for every $s>1$ (margin $0.045$--$0.093$). Since $|\cdot|$ is
$1$-Lipschitz, $F(s)\le F_{\mathrm{gauss}}(s)+\sqrt{2/\pi}\int_0^1\bigl(\mathbb E|Z|-\mathbb E|G|\bigr)\,du$, and
the Gaussian excess has the exact Kluyver/Hankel form $\mathbb E|Z|-\mathbb E|G|=\int_0^\infty(M_{\mathrm g}(t)-M(t))\,t^{-2}\,dt$,
$M(t)=\tfrac1{2\pi}\int_0^{2\pi}\prod_{\mathrm{even}}J_0(a_\ell t\cos\theta)\prod_{\mathrm{odd}}J_0(a_\ell t\sin\theta)\,d\theta$
(and $M_{\mathrm g}$ with $J_0$ replaced by the Gaussian factor). The excess is positive, sourced entirely by
the Bessel rings $t>j_{0,1}=2.405$ (on $[0,j_{0,1}]$, $J_0(z)\le e^{-z^2/4}$). The certified bound is the
fourth-cumulant estimate
\[
  \mathbb E|Z|-\mathbb E|G|\ \le\ K_\star\,\frac{\sum_\ell a_\ell^4}{\Sigma_2^{3/2}},\qquad
  K_\star=\frac{h(1)-\sqrt\pi/2}{2}=0.035932,
\]
\emph{for the profile's balanced configurations}, sharp at the equal two-copy point ($a=b$, the
$\beta_{\mathrm{odd}}$ configuration). The restriction is essential and structural: the live set
$L(s)=\{\ell:|u+\ell|<s\}$ is an interval of \emph{consecutive} integers, so its even and odd parts differ in
size by at most one (\emph{balanced} channels); a general \emph{imbalanced} vector need not obey the
bound---one even copy against five odd exceeds $K_\star$ by $\approx9\%$
(\texttt{verify\_dagger\_extremal.py})---but no such configuration occurs in the profile. The
\emph{two-copy core} $R(1,\rho)\le K_\star$ ($\rho=b/a$, equality iff $\rho=1$) is now \emph{proved}: the
symmetry $R(1,\rho)=R(1,1/\rho)$ forces $R'(1)=0$, so with $\Phi=K_\star P-(\mathbb E|Z|-\mathbb E|G|)$ one has
$\Phi(1)=\Phi'(1)=0$, $\Phi''\ge0$ on $[0.6,1]$ and $\Phi\ge0$ on $(0,0.6]$ (validated interval arithmetic),
whence $\Phi(\rho)=\int_\rho^1(x-\rho)\Phi''\,dx\ge0$ (\texttt{verify\_dagger\_extremal2.py}). The
$\ge3$-copy balanced case stays \emph{validated} (no balanced configuration beats the pair) and---since $R$
is \emph{not} Schur-monotone---does \emph{not} reduce to the two-copy core.

Feeding the bound through the $u$-integral gives the majorant
\[
  B(s)=F_{\mathrm{gauss}}(s)+\sqrt{2/\pi}\,K_\star\int_0^1\Sigma_2^{-3/2}\textstyle\sum_\ell a_\ell^4\,du\ \ge\ F(s),
\]
which clears $\beta_{\mathrm{odd}}$ on $[1.005,\infty)$, where the admissible constant
$K_{\max}(s)\ge0.0363>K_\star$ (the largest $K$ for which $B<\beta_{\mathrm{odd}}$).
On the thin sliver $(1,1.005]$ the majorant is \emph{not} used---there $K_{\max}$ dips to $0.0338<K_\star$, so
$B$ alone would not clear $\beta_{\mathrm{odd}}$---and instead the \emph{proved} variance floor $F\le F_K$
(Lemma~\ref{lem:varfloor}) gives $F\le F_2<\beta_{\mathrm{odd}}$ directly (the live set is essentially two
copies, margin $\ge0.003$; \texttt{verify\_dagger\_window.py}). So $\max_{s\ge1}F(s)=F(1)=\beta_{\mathrm{odd}}$
modulo the single $\ge3$-copy balanced expected-modulus inequality. The off-the-shelf routes fail
quantitatively: the $1$-D Wasserstein/Esseen CLT bound (Esseen's sharp constant $\tfrac12$) overshoots the
margin tenfold because $\sum a_\ell^3/\sigma^2$ stays $\Theta(1)$; the crude Bessel-ring tail bound also
overshoots ($\approx1.7K_\star$, the excess being a \emph{signed} ring sum whose cancellation a modulus bound
destroys), so the residual genuinely needs the signed tail or a multivariate concavity argument.

\noindent What is now \emph{proved}, for every $s>1$: the live Debye phases are jointly equidistributed with
an explicit $N^{-1/2}$ power saving (Theorem~\ref{thm:joint-equi}), resting only on the linear independence of
the curvature kernels $g_\ell$ (distinct poles); and this transfers \emph{uniformly in $u$} to the $L^1$
limit (Theorem~\ref{thm:uniform-transfer}), so $\|K_N(\cdot,sN/2)\|_1=(F(s)+o(1))\sqrt N$ holds for every fixed
$s>1$ unconditionally---the former residue (i) (the band-edge strips contributing $O(\varepsilon^{3/4})$ and
the coalescing points $u=-(\ell+\ell')/2$ handled by the cubic test, $g_\ell'-g_{\ell'}'\ne0$, with no triple
coincidence) is closed. There also hold the variance-floor inequality $F\le F_K$ and the tail limit
$\sqrt\pi/2<\beta_{\mathrm{odd}}$ (Lemma~\ref{lem:varfloor}). \emph{A single residual estimate}, validated
numerically, now separates this from an unconditional $\limsup_N B_N/\sqrt N=\beta_{\mathrm{odd}}$: the
$\ge3$-copy case of the balanced Gaussian-excess inequality $\mathbb E|Z|-\mathbb E|G|\le K_\star\,\Sigma_2^{-3/2}\sum_\ell a_\ell^4$
above (the former residue (ii)). Its \emph{two-copy core} $R(1,\rho)\le K_\star$ is \emph{proved}
(\texttt{verify\_dagger\_extremal2.py}), the sliver $s\in(1,1.005]$ is covered by the proved variance floor
$F\le F_2<\beta_{\mathrm{odd}}$, and the $\ge3$-copy balanced configurations are validated never to beat the
pair; but $R$ is not Schur-monotone, so this last step does not reduce to the two-copy core. The two natural
routes are now mapped, and neither closes it cleanly: $R$ is \emph{not} concave on the balanced simplex (its
multi-copy critical points are saddles whose ascent directions run toward the imbalanced violators, so the
balance must be used \emph{actively}), while the signed Bessel-ring tail---a two-frequency alternating sum
($\omega=1,\sqrt2$ with explicit amplitudes; crude modulus bounds overshoot to $1.7K_\star$), magnitude-%
decreasing as balanced pairs are added---is understood but not yet bounded rigorously over \emph{all}
balanced configurations. A finer split narrows the gap to one sub-family: in amplitude coordinates
$\Phi=K_\star P-(\mathbb E|Z|-\mathbb E|G|)$ is \emph{convex} at the pair (tangent eigenvalues
$+0.38,\dots,+6.0$), so the two-copy core extends to a certified near-pair basin (concentration
$\lambda\ge0.82$), and configurations with $\ge2$ comparable copies in \emph{each} channel fall to the crude
modulus bound; and the \emph{singleton-channel} family (one channel a single copy, $\lambda\in[0.65,0.82)$,
$R\le0.91$) is now bounded below $K_\star$ \emph{pointwise} by a rigorous three-region split: a cumulant
majorant on $[0,1.5]$, an Arb \emph{Taylor model} of the excess on $[1.5,12]$ (the cancellation kept inside
each polynomial box, which naive ball arithmetic destroys), and a capped-modulus tail on $[12,\infty)$
(worst-config Arb upper bound $0.991\,K_\star$; \texttt{verify\_dagger\_singleton.py},
\texttt{verify\_dagger\_decomp.py}). Every region of the balanced simplex is thereby bounded below $K_\star$ at
each evaluated configuration; what is \emph{not} yet a continuum theorem is the uniform cover between grid
nodes---a $0.01$-wide interval-amplitude box inflates the cumulant majorant fourfold, so the cover rests on a
grid-Lipschitz standing assumption, not a closed interval enclosure. A multivariate Taylor model in the
amplitudes (degree~$2$ with a Lagrange remainder) does remove that fourfold ball inflation---the per-box lift
is implemented and the true ratio stays $\le0.91$ throughout (\texttt{verify\_dagger\_continuum.py})---and the
checkpointed runner \texttt{run\_dagger\_continuum\_tiling.py} now closes full-resolution boxes wherever the
Taylor margin is adequate. The equal-amplitude singleton zone near $a_1=a_3\simeq0.73$ resisted not because
the inequality is tight there---the true ratio is $\le0.88$, a margin $\simeq0.12$---but because the
\emph{signed} $\theta$-ball middle's rigorous enclosure radius blows up past $t=12$ (it scales like
$\mathrm{amp}\cdot t/n_\theta$, which is exactly why the band middle stops at $T_2=12$). A $\theta$-Taylor
$+$ $t$-collocation enclosure restores it for $t\ge8$ (\texttt{middle\_far\_point}: $\theta$ is a Taylor
series variable so the fast oscillation is captured analytically, while $t$ is sampled at \emph{points}, so
the $0F1$ Bessel series is never evaluated on an interval---the only thing that makes its catastrophic
cancellation inflate the radius); it is validated against the reference enclosure (containment, radius
$\sim10^{-7}$; \texttt{verify\_middle\_far.py}). With the split $\mathrm{mid}[1.5,8]+\mathrm{far}[8,24]+\mathrm{tail}[24,\infty)$
through \texttt{prove\_box($\dots$,\,far\_slices=$\dots$)} the previously-failing boxes certify
($(0.70,0.70)\to0.948$, $(0.73,0.73)\to0.943$, $(0.76,0.76)\to0.922$, the worst true value sitting at
$a\simeq0.70$). The full singleton-band cover is the adaptive, node-parallel, checkpointed tiling

$\limsup_N B_N/\sqrt N=\beta_{\mathrm{odd}}$ at the certificate's validated-enclosure standard. The honest status is that the multi-copy region is
reduced from a black-box Weyl-sum/anti-flatness problem to one sharply localized expected-modulus
estimate---the entire $N\to\infty$ reduction (joint equidistribution and its uniform transfer to the $L^1$

Remark~\ref{rem:bn-parity}. The closest
published result, an $L^4$ bound for the same discrete chirp, stops short of the $L^1$
constant~\cite{FuRenWang}.

\subsection{Bessel kernel and Poisson summation}
On the infinite chain $\Z$ the propagator kernel follows from the symbol by a contour integral,
\begin{equation}\label{eq:bessel}
  K_\infty(n,t)=\frac1{2\pi}\int_{-\pi}^{\pi}\eu^{\mathrm in\theta}\,\eu^{-\mathrm it(2-2\cos\theta)}\,d\theta
  =\eu^{-2\mathrm it}\,\mathrm i^{\,n}J_n(2t),
\end{equation}
by the Jacobi--Anger identity $\frac1{2\pi}\int\eu^{\mathrm i(n\theta+x\cos\theta)}\,d\theta=\mathrm i^nJ_n(x)$~\cite{Watson}.
Thus $|K_\infty(n,t)|=|J_n(2t)|$: a ballistic light cone $|n|\approx2t$ with an Airy front of height
$\max_n|J_n(2t)|\sim0.6748\,(2t)^{-1/3}$ (Figure~\ref{fig:bessel}). Unitarity gives $\sum_n|K_\infty|^2=1$,
while the $\ell^1$ mass $\sum_n|J_n(2t)|\sim c\sqrt t$ grows---the infinite-chain shadow of
$B_N\sim\sqrt N$. The ring kernel is the Poisson periodization
\begin{equation}\label{eq:poisson}
  K_N(j,t)=\sum_{\ell\in\Z}K_\infty(j+\ell N,t),
\end{equation}
verified to machine precision; the large wave survives only through this finite-$N$ folding, since on
$\Z$ the front escapes and the peak decays.

\subsection{Arithmetic dynamics: revivals and Gauss sums}
The recurrence structure of $\eu^{-\mathrm i t\LN}$ is arithmetic. Since $2\cos(2\pi/N)$ is rational
only for $N\in\{1,2,3,4,6\}$ (Niven's theorem~\cite{Niven}; for general $r$, $2\cos(2\pi r/N)$ is
rational only when $r/N$ reduces to such a denominator), the tight-binding spectrum is generically
incommensurate: $\eu^{-\mathrm i t\LN}$ has \emph{no finite revival period}, and the return amplitude
$|K_N(0,t)|$ is almost-periodic, approaching $1$ along a wandering sequence of near-revival times
(\texttt{verify\_revivals\_gauss.py}). In the long-wave limit $\lambda_r\approx(2\pi r/N)^2$ the model
approaches the \emph{quadratic} dispersion of a free particle, where the dynamics becomes exactly
arithmetic: at the Talbot times $t=2\pi s/q$ (in the rescaled time $\tau=(2\pi/N)^2 t$ of this quadratic
limit) the propagator kernel is a quadratic Gauss sum, and for
odd $q$ with $\gcd(a,q)=1$,
\[
  \Bigl|\sum_{n=0}^{q-1}\eu^{2\pi\mathrm i a n^2/q}\Bigr|=\sqrt q,
\]
yielding an exact fractional revival of uniform amplitude $|U_{jk}|=1/\sqrt q$ (and a full revival
$U(2\pi)=I$). The large wave thus inhabits the almost-periodic, Diophantine regime of the lattice, of
which the discrete Talbot effect and Gauss-sum revivals are the exactly solvable continuum shadow.

\subsection{Dirichlet segment: a logarithmic site-amplitude law}\label{sub:filimonov}
The sharpest large-wave statement for the discrete Schr\"odinger equation is Filimonov's~\cite{F23}, and
it lives on the \emph{open} (Dirichlet) chain rather than the ring. With on-site value
$\varepsilon_j\equiv-2$ and pinned ends $z_0=z_N=0$,
\[
  \mathrm i\dot z_j=z_{j+1}-2z_j+z_{j-1},\qquad z_j(0)=\alpha_j,\qquad 1\le j\le N-1,
\]
the Dirichlet sine basis diagonalizes the flow and the solution is explicit,
\begin{equation}\label{eq:fil-sol}
  z_j(t)=\eu^{2\mathrm it}\sum_{k=1}^{N-1}a_k\sin\frac{\pi kj}{N}\,\eu^{-2\mathrm i\omega_k t},
  \qquad \omega_k=\cos\frac{\pi k}{N},\qquad a_k=\frac2N\sum_{r=1}^{N-1}\alpha_r\sin\frac{\pi kr}{N}.
\end{equation}
The simple spectrum $\{\omega_k=\cos(\pi k/N)\}$ replaces the twofold-degenerate ring frequencies. For the
uniform initial state $\alpha_j\equiv1$ the coefficient sum is the imaginary part of a geometric series,
$\sum_{r=1}^{N-1}\sin(\pi kr/N)=\cot(\pi k/2N)$ for odd $k$ and $0$ for even $k$, so
\begin{equation}\label{eq:fil-ak}
  a_k=\frac2N\cot\frac{\pi k}{2N}\ \ (k\ \text{odd}),\qquad a_k=0\ \ (k\ \text{even}).
\end{equation}
The pointwise ceiling is $d_j^N:=\sum_{k=1}^{N-1}\bigl|a_k\sin(\pi kj/N)\bigr|$, with $|z_j(t)|\le d_j^N$
for all $t$ by the triangle inequality.

\begin{theorem}[Filimonov~\cite{F23}]\label{thm:filimonov}
For $\alpha_j\equiv1$ and $N=2^m$, $\ \sup_t|z_j(t)|=d_j^N$, and
\begin{equation}\label{eq:fil-lnj}
  \lim_{N\to\infty}d_j^N=C\ln j+O(1).
\end{equation}
\end{theorem}

\noindent Thus the Schr\"odinger amplitude at the \emph{observation site} $j$ grows without bound
(logarithmically) as $j\to\infty$---a large wave at the far end of the chain, approached along the
relatively dense set of near-revival times (the modulus is almost-periodic, \cite{Bohr}). Two refinements
complete the picture (\texttt{verify\_filimonov\_schrodinger.py}).

\emph{The constant is not new.} It is the value Myshkis--Filimonov established for the \emph{wave}
problem: their segment ceiling satisfies $\zeta_j^N\to\tfrac{4}{\pi^2}\ln j+O(1)$
\cite[Thm.~2]{MF03}, and the Schr\"odinger ceiling \eqref{eq:fil-ak} coincides with it term for term
(both equal $\tfrac2N\sum_{k\ \mathrm{odd}}\bigl|\cot\tfrac{\pi k}{2N}\sin\tfrac{\pi kj}{N}\bigr|$). The same
limit is immediate here: as $N\to\infty$ the ceiling \eqref{eq:fil-ak} becomes a Riemann sum for
\begin{equation}\label{eq:fil-int}
  d_j^\infty=\frac2\pi\int_0^{\pi/2}\cot\theta\,\bigl|\sin 2j\theta\bigr|\,d\theta,
\end{equation}
and since $\cot\theta\sim\theta^{-1}$ near $0$ while $\langle|\sin|\rangle=2/\pi$, the integral grows like
$\tfrac2\pi\cdot\tfrac2\pi\ln j$, giving
\begin{equation}\label{eq:fil-C}
  C=\frac{4}{\pi^2}\approx0.4053 .
\end{equation}
The genuinely new content of Theorem~\ref{thm:filimonov} is the Schr\"odinger \emph{attainment}
$\sup_t|z_j|=d_j^N$ \cite{F23}, not the constant. The fit $d_j^N=(4/\pi^2)\ln j+O(1)$ holds to
$R^2=1.0000$ at $N=2^{12}$ (Figure~\ref{fig:filimonov}).

\emph{Independence and the scope of Theorem~\ref{thm:filimonov}.} The equality $\sup_t|z_j|=d_j^N$ rests
on two facts: the sign-pairing $\omega_{N-k}=-\omega_k$ together with the elementary identity
$\max(|x+y|,|x-y|)=|x|+|y|$ lets each pair reach $|a_k|+|a_{N-k}|$, and the distinct $|\omega_k|$ must be
$\Q$-independent so that all pairs align simultaneously (Kronecker). This independence---the cosine
analogue of Theorems~\ref{thm:prime}--\ref{thm:two}, via $2\cos(\pi k/N)=\zeta_{2N}^k+\zeta_{2N}^{-k}$ in
the \emph{plus}-eigenspace---has full rank $\lfloor(N-1)/2\rfloor$ exactly for $N$ prime or $N=2^m$
(\texttt{verify\_filimonov\_schrodinger.py}), which \emph{suffices} for $\sup_t|z_j|=d_j^N$ at every site
$j$; so Filimonov's $N=2^m$ extends to prime $N$. Full rank is not \emph{necessary}, however: composite
$N$ can still saturate at individual Dirichlet sites---numerically $\sup_t|z_j|=d_j^N$ already at $j=1$
for $N=6,9,10,12$, the alignment there needing only the active frequencies and the sign-pairing---so a
complete classification of this site-wise segment saturation is open. This is separate from the ring
classification Theorem~\ref{thm:classification}, which is settled.

\emph{Three laws on one Laplacian.} The discrete Laplacian thus carries several distinct large-wave
asymptotics, one per dynamics and functional (Table~\ref{tab:laws}): $A_N\sim\tfrac1\pi\ln N$ (ring,
growth in the system size $N$), $B_N\sim\sqrt N$ (ring, $\ell^\infty\!\to\!\ell^\infty$ operator norm),
and $d_j^N\sim\tfrac4{\pi^2}\ln j$ (Dirichlet segment, growth in the observation site $j$).

\section{Continuum analogue and the step initial condition}

Expanding the symbol for small wavenumbers, $\omega(\theta)=2\sin(\theta/2)=\theta-\theta^3/24+O(\theta^5)$,
recovers to leading order the continuum wave equation $u_{tt}=u_{xx}$ ($\omega=\theta$). A finite Taylor
truncation improves the long-wave fit but diverges from the exact branch near the band edge
$\theta=\pi$. The \emph{two-point Pad\'e} approximant of Andrianov--Awrejcewicz~\cite{AAW}, obtained from the
regularized model $m(1-a^2h^2\partial_x^2)u_{tt}=ch^2u_{xx}$ with $a^2=\tfrac14-\pi^{-2}\approx0.1487$,
yields
\begin{equation}\label{eq:pade}
  \omega_{\mathrm{Pad\acute{e}}}(\theta)=\frac{\theta}{\sqrt{1+a^2\theta^2}},
\end{equation}
which matches the exact dispersion at \emph{both} ends---unit slope at $\theta=0$ and
$\omega=2$ at $\theta=\pi$ (indeed $\pi/\sqrt{1+a^2\pi^2}=2$)---with error below $3\%$ in between
(Figure~\ref{fig:dispersion}). Nonetheless no continuum surrogate reproduces the large wave: as
Section~\ref{sec:schr} shows through the decaying Bessel kernel, the effect is \emph{absent} in the
continuum/infinite limit---it is intrinsically a finite-size, discrete phenomenon driven by the exact
arithmetic of the sampled frequencies $\{2\sin(\pi r/N)\}$.

\begin{figure}[ht]\centering
\includegraphics[width=.72\textwidth]{fig_dispersion.pdf}
\caption{Dispersion relations. The exact lattice branch $2\sin(\theta/2)$ is matched at both ends by
the two-point Pad\'e \eqref{eq:pade}, whereas the continuum $\theta$ and the Taylor truncation
$\theta-\theta^3/24$ diverge at the band edge $\theta=\pi$ ($\bullet$, $\omega=2$).}\label{fig:dispersion}
\end{figure}

For the step displacement $u_j(0)=1$ ($1\le j\le N-1$), $u_0(0)=0$, $\dot u(0)=0$, the exact supremum
is, by Bohr's theorem (for $N=2^m$), $\sup_t\max_j u_j(t)=\bar\alpha+\max_j\sum_r|A_{r,j}|\approx2$ for
all $N$, where $\bar\alpha=(N-1)/N$ and $A_{r,j}$ are the modal amplitudes. Earlier numerical work
reported peaks $1.617\to1.397$ for $N=32,\dots,1024$; these are finite-time scan values, and their
\emph{decrease} with $N$ is the expected worsening of the scan undershoot for an almost-periodic
supremum (the true maximum is only approached, at near-recurrence times). The present analysis thus
reproduces the effect and corrects the value to $\approx2$ (\texttt{verify\_thesis\_peaks.py}).

\section{Statistics: density of states and value distribution}\label{sec:stats}

The spectrum and the large wave both admit clean probabilistic descriptions, verified in
\texttt{verify\_statistics.py}.

\emph{Density of states.} As $N\to\infty$ the eigenvalues $\lambda_r=4\sin^2(\pi r/N)$---the pushforward
of the uniform law by $\lambda(x)=2-2\cos2\pi x$---converge to the \textbf{arcsine law} on $[0,4]$,
\begin{equation}\label{eq:arcsine}
  \rho(\lambda)=\frac1{\pi\sqrt{\lambda(4-\lambda)}},\qquad F(\lambda)=\frac2\pi\arcsin\frac{\sqrt\lambda}{2}.
\end{equation}
This is the change of variables: $\tfrac{d\lambda}{dx}=4\pi\sin2\pi x=2\pi\sqrt{\lambda(4-\lambda)}$ (since
$\sin2\pi x=\tfrac12\sqrt{\lambda(4-\lambda)}$), and each $\lambda\in(0,4)$ has \emph{two} preimages in
$[0,1]$, so $\rho(\lambda)=2/(d\lambda/dx)=1/(\pi\sqrt{\lambda(4-\lambda)})$---the classical density of
states of the one-dimensional discrete Laplacian (Figure~\ref{fig:dos}); the frequencies have
$\rho(\omega)=2/(\pi\sqrt{4-\omega^2})$ on $[0,2]$. Both fit to Kolmogorov--Smirnov distance $O(N^{-1})$
(a deterministic quantile grid: the empirical CDF lies within $2/N$ of the limit, the $1/N$ jump floor).

\emph{Value distribution.} For prime $N$ the impulse response $u_0(t)$ is almost-periodic with
$\Q$-independent frequencies, so by the Bohr--Jessen theorem its values possess a limiting distribution
(Figure~\ref{fig:dos}, right): symmetric, mean $0$, and variance $\mathrm{Var}_t\,u_0=\tfrac{N^2-1}{12N^2}\to\tfrac1{12}$.
The variance is exact: with $u_0(t)=\tfrac1N\sum_r\omega_r^{-1}\sin(\omega_r t)$ and the time-average
$\langle\sin(\omega_r t)\sin(\omega_s t)\rangle=\tfrac12$ when $\omega_s=\omega_r$ (i.e.\ $s\in\{r,N-r\}$)
and $0$ otherwise, the degeneracy makes each cross-term reinforce the diagonal, so
$\langle u_0^2\rangle=\tfrac1{N^2}\sum_{r}\omega_r^{-2}=\zeta_{\LN}(1)/N^2=\tfrac{N^2-1}{12N^2}$. The law is
\emph{platykurtic} (kurtosis $\approx2.4<3$) because the lowest mode dominates
(weight $1/\omega_1\sim N$, a single arcsine sinusoid). The large wave is therefore \emph{not} a heavy
tail of the typical value but the rare near-alignment supremum (Figure~\ref{fig:largewave}): a spike that
climbs to nearly $U_N$ standing far above the thin-tailed bulk of size $\mathrm{RMS}\approx1/\sqrt{12}$,
$A_N\sim\tfrac1\pi\ln N$.

\section{Higher-dimensional tori: the large wave is one-dimensional}\label{sec:dim}

On the $d$-dimensional torus $(\Z_N)^d$ the Laplacian is block-circulant-with-circulant-blocks (BCCB),
diagonalized by the $d$-dimensional DFT with eigenvalues
$\lambda_{\mathbf k}=\sum_{i=1}^d4\sin^2(\pi k_i/N)$ and $\omega_{\mathbf k}=\sqrt{\lambda_{\mathbf k}}$.
The triangle inequality gives, unconditionally,
$A_N^{(d)}\le U_N^{(d)}=N^{-d}\sum_{\mathbf k\ne\mathbf0}\omega_{\mathbf k}^{-1}$. Near the origin
$\omega_{\mathbf k}\sim(2\pi/N)|\mathbf k|$, so $N^d U_N^{(d)}$ mirrors the infrared lattice sum
$\sum 1/|\mathbf k|$, i.e.\ the integral $\int d^dk/|\mathbf k|$: logarithmically divergent for $d=1$
(giving $U_N\sim\frac1\pi\ln N$) but \emph{convergent} for $d\ge2$. Hence (Figure~\ref{fig:dimension})
\begin{equation}\label{eq:dim}
  U_N^{(d)}=\Theta(\ln N)\ \ (d=1),\qquad U_N^{(d)}=O(1)\ \ (d\ge2):
\end{equation}
the velocity Green's function---and with it the large-wave amplification---is \emph{uniformly bounded}
on the $2$- and $3$-torus. The large wave is intrinsically one-dimensional, a sharp dimensional
threshold set by the integrability of $1/\omega$ (\texttt{verify\_torus\_dimension.py}).

\medskip\noindent\textbf{Spectral dimension.} The threshold is not metric but \emph{spectral}. For
vertex-transitive (Fourier-flat) graph families the local ceiling reduces to the normalized spectral sum
$U_G=|V|^{-1}\sum_{\lambda_r>0}\lambda_r^{-1/2}$ ($\lambda_r$ the Laplacian spectrum,
$\omega_r=\sqrt{\lambda_r}$); for a general graph the ceiling at $(j,k)$ is
$\sum_{\lambda_r>0}|\phi_r(j)\phi_r(k)|\lambda_r^{-1/2}$ and also reflects eigenvector localization through
the local spectral measure, not the global density of states alone.
If the integrated density of states obeys $N(\lambda)\sim\lambda^{d_s/2}$ near $0$---defining the
\emph{spectral dimension} $d_s$---then $\rho(\lambda)\sim\lambda^{d_s/2-1}$ and
$U_G\sim\int_0\lambda^{-1/2}\rho(\lambda)\,d\lambda\sim\int_0\lambda^{(d_s-3)/2}\,d\lambda$, which
converges at the origin iff $d_s>1$. Hence the \emph{ceiling} $U_G$ grows logarithmically precisely at
the critical spectral dimension $d_s=1$, is bounded for $d_s>1$, and is power-law for $d_s<1$; the
$d$-torus is the case $d_s=d$. For $d_s>1$ this rigorously \emph{forbids} a large wave
($A_N\le U_G=O(1)$); at $d_s=1$ the ceiling permits one and \emph{is} reached on the ring
($A_N\sim\tfrac1\pi\ln N$, Theorem~\ref{thm:order}), the same question for a general quasi-1D lattice being
the natural next case. The criterion is geometric rather than metric: any
\emph{quasi-one-dimensional} lattice---a ladder $C_N\times C_2$ or a tube $C_N\times C_w$ of bounded
width $w$---still has $d_s=1$ and retains a logarithmically divergent ceiling, now with reduced prefactor
$U_G\sim\tfrac1{\pi w}\ln N$ (only one of the $w$ transverse channels is gapless, Figure~\ref{fig:specdim}).
Bounded transverse width does not destroy the concentration; a genuine second extended dimension does
(\texttt{verify\_spectral\_dimension.py}).

The order argument is in fact graph-agnostic: the same Kronecker alignment proves $\Theta(\ln N)$ for any
family meeting two conditions.

\begin{proposition}[order criterion at $d_s=1$]\label{prop:order-criterion}
Let $\{G_N\}$ be finite graphs whose diagonal velocity Green's function is
$G_N(t)=\sum_r b_r\sin(\omega_r t)$ with $b_r>0$ and ceiling $U_N=\sum_r b_r$. Suppose
\textup{(A)} $U_N\sim c\,\ln N$ ($c>0$; the $d_s=1$ logarithmic ceiling), and \textup{(B)} some
$\Q$-independent subset $S$ of the frequencies carries $\sum_{r\in S}b_r\ge(\tfrac12+\delta)U_N$ for a fixed
$\delta>0$. Then $A_N:=\sup_t\max_{\mathrm{node}}|G_N(t)|$ obeys $2\delta\,U_N\le A_N\le U_N$, hence
$A_N=\Theta(\ln N)$.
\end{proposition}

\begin{proof}
$A_N\le U_N$ since $|\sin|\le1$. For the lower bound, by Kronecker the $\Q$-independent phases
$\{\omega_r t\}_{r\in S}$ are simultaneously $\to\pi/2$ at suitable $t$; there $G_N(t)\ge\sum_{r\in S}b_r
-\sum_{r\notin S}b_r=2\sum_{r\in S}b_r-U_N\ge2\delta U_N$ (the complement bounded below by $\sin\ge-1$).
\end{proof}

\noindent The ring and Dirichlet segment are the instances where Lemma~\ref{lem:prefix} supplies $S$ (the
length-$\tfrac12\varphi(2N)$ prefix, carrying $(1-o(1))U_N$, so $\delta\to\tfrac12$). For a quasi-1D lattice
(A) holds with $c=1/(\pi w)$, while (B)---a long $\Q$-independent prefix---is the per-family analogue of the
palindromic lemma: e.g.\ the next-nearest-neighbour ring, with
$\omega_r=2\sin(\tfrac{\pi r}N)\sqrt{2+\cos\tfrac{2\pi r}N}$, is $\Q$-independent at $N=15$ (so it
\emph{saturates}, $A_N=U_N$) yet carries the degeneracy $\omega_3=\omega_6=2$ at $N=12$
(\texttt{verify\_quasi1d\_order.py}). For this family the \emph{degeneracy layer} of (B) can be settled
completely.

\begin{proposition}[NNN degeneracy law]\label{prop:nnn}
For the next-nearest-neighbour ring ($g=\tfrac12$), two distinct frequencies coincide,
$\omega_a=\omega_b$ with $a<b\le\lfloor N/2\rfloor$, \emph{if and only if} $4\mid N$ and
$(a,b)=(N/4,\,N/2)$, where both equal $2$.
\end{proposition}

\begin{proof}
With $x=\cos\tfrac{2\pi a}N$ one has $\omega_a^2=f(x)=2(1-x)(2+x)$, and
$f(x_a)-f(x_b)=-2(x_a-x_b)(1+x_a+x_b)$. Since $\cos$ is injective on $(0,\pi]$ and $a\ne b$, a collision
forces $x_a+x_b=-1$, i.e.\ the vanishing of the length-$6$ positive sum
$2+\zeta^a+\zeta^{-a}+\zeta^b+\zeta^{-b}$ of $N$-th roots of unity. By the Conway--Jones classification of
rational cosine relations~\cite{CJ}, the rational-angle solutions of $\cos A+\cos B=-1$ are
$\{A,B\}=\{\pi/2,\pi\}$ and $A=B=2\pi/3$ (excluded by $a\ne b$); hence $a=N/4$, $b=N/2$
(\texttt{verify\_nnn\_collisions.py}: confirmed \emph{exactly}, in integer arithmetic modulo $\Phi_N$, for
all $N\le200$). \end{proof}

\noindent The law generalises to every rational coupling: for next-nearest weight $g>0$ the symbol
factorises as $\omega^2=(1-x)\bigl(2+4g(1+x)\bigr)$, $x=\cos\tfrac{2\pi a}N$, so distinct frequencies
collide iff $\cos\tfrac{2\pi a}N+\cos\tfrac{2\pi b}N=-\tfrac1{2g}$---again a rational cosine relation
classified by~\cite{CJ}. For $g=1$ the solutions are $\{\pi/3,\pi\}$, $\{2\pi/5,4\pi/5\}$,
$\{\pi/2,2\pi/3\}$, so collisions occur exactly at $(N/6,N/2)$, $(N/5,2N/5)$, $(N/4,N/3)$ when $6$, $5$,
$12$ divide $N$ (verified exactly for $N\le120$); for $g=\tfrac14$ the required value $-2$ forces
$A=B=\pi$, and there are \emph{no} collisions for any $N$. The degeneracy layer of condition (B) is thus
settled for the whole one-parameter NNN family. Beyond collisions, a $100$-digit \textsc{pslq} search finds
no further rational relation among the NNN ($g=\tfrac12$) frequencies for any $5\le N\le40$; dropping the
duplicate when $4\mid N$ then leaves a $\Q$-independent set carrying the full ceiling, so on this evidence
condition (B)---and with it $A=\Theta(\ln N)$---holds throughout the family. Deciding (B) for a
\emph{general} $d_s=1$ lattice remains open; the criterion isolates exactly what must be shown.

For one natural class, though, (B) is free---whenever the ring sits inside the lattice as a factor.

\begin{corollary}[order on ring products]\label{cor:product}
For any fixed connected graph $H$, the large wave on the Cartesian product $C_N\,\square\,H$ satisfies
$A=\Theta(\ln N)$.
\end{corollary}

\begin{proof}
The Laplacian of $C_N\,\square\,H$ has eigenvalues $\lambda_r+\mu_h$, with $\lambda_r=2-2\cos\tfrac{2\pi r}N$
on $C_N$ and $\mu_h$ on $H$. Only the gapless branch $h=0$ ($\mu_0=0$, $H$-constant eigenvector
$w_0=|H|^{-1/2}\mathbf1$) has $\omega\to0$, so the ceiling $U\sim\tfrac1{\pi|H|}\ln N$ is carried by it
while the gapped branches add $O(1)$---condition (A). The $h=0$ frequencies are exactly the ring
frequencies $2\sin(\pi r/N)$, whose first $\tfrac12\varphi(2N)$ are $\Q$-independent by
Lemma~\ref{lem:prefix} and contribute $(1-o(1))U>\tfrac12U$---condition (B). Proposition~\ref{prop:order-criterion}
then gives $A=\Theta(\ln N)$ (\texttt{verify\_product\_order.py}: the prefix bound $2L_{\mathrm{pre}}-U$ is
positive and grows like $\ln N$ for $H=K_2,P_3$). \end{proof}

\noindent So the order law survives multiplication by any bounded ``cross-section'': ladders, prisms,
tubes $C_N\,\square\,H$ all carry the $\Theta(\ln N)$ large wave, the embedded ring supplying the
arithmetic. The same embedded-ring mechanism reaches beyond products.

\begin{corollary}[M\"obius ladder]\label{cor:mobius}
The large wave on the M\"obius ladder $M_N$---the circulant $C_{2N}(\{1,N\})$ on $2N$ vertices, \emph{not}
a Cartesian product---satisfies $A(M_N)=\Theta(\ln N)$.
\end{corollary}

\begin{proof}
The Laplacian eigenvalues are $\Lambda_r=2\bigl(1-\cos\tfrac{\pi r}{N}\bigr)+\bigl(1-(-1)^r\bigr)$,
$r=0,\dots,2N-1$. The even branch $r=2j$ gives $\Lambda_{2j}=2\bigl(1-\cos\tfrac{2\pi j}{N}\bigr)=
4\sin^2(\pi j/N)$---\emph{exactly} the ring $C_N$ spectrum---while the diameter chord gaps the odd branch by
$1-(-1)^r=2$. So $\omega\to0$ only on the even branch, which carries the ceiling $U\sim\tfrac1{2\pi}\ln N$
(condition~A); its frequencies are the ring frequencies $2\sin(\pi j/N)$, whose first $\tfrac12\varphi(2N)$
are $\Q$-independent by Lemma~\ref{lem:prefix} (exact integer-rank certificate, e.g.\ rank
$128=\tfrac12\varphi(512)$ at $N=256$), contributing $(1-o(1))U$ (condition~B).
Proposition~\ref{prop:order-criterion} gives $A(M_N)=\Theta(\ln N)$
(\texttt{verify\_kjump\_order.py}). \end{proof}

\begin{remark}[multi-jump circulants saturate]\label{rem:kjump}
For a $k$-jump circulant $C_N(S)$, $|S|\ge2$, the frequencies are the nested radicals
$\omega_r=\bigl(\sum_{s\in S}4\sin^2(\pi sr/N)\bigr)^{1/2}$, and condition~(A) holds with slope
$1/(\pi\sqrt{\sum_{s\in S}s^2})$. Distinct frequencies coincide only along exact Conway--Jones congruences
(for $S=\{1,2,3\}$: a collision iff $4\mid N$, $6\mid N$, or $7\mid N$; for $S=\{1,3\}$: iff $8\mid N$ or
$5\mid N$), and---unlike the single sines of the ring, which carry the composite mod-$4$ obstruction of
Theorem~\ref{thm:classification}---the \emph{distinct} radicals appear to satisfy no further $\Q$-relation
(\textsc{pslq} to $250$ digits, the spurious relation norm diverging with precision). Numerically the
multi-jump circulant then \emph{saturates}, $A_N=U_N$, for every $N$ off its collision congruences: the
radical structure destroys the cyclotomic linear relations, so composite $N$ is no longer an obstruction
(\texttt{verify\_kjump\_order.py}). For the nearest case $S=\{1,2\}$ this is no longer only numerical: the
$\Q$-independence of $\{\sqrt{\alpha_r}\}$, $\alpha_r=4-2\cos\tfrac{2\pi r}p-2\cos\tfrac{4\pi r}p\in
\Q(\zeta_p)$, is \emph{proved exactly} by the Besicovitch--Kummer criterion---no nonempty subset product
$\prod_{r\in T}\alpha_r$ is a square in $\Q(\zeta_p)$ (checked over the cyclotomic field, not in
floating point)---for every collision-free prime $p\le31$, so $A_p=U_p$ there is a theorem
(\texttt{exact/pari/kjump\_kummer.gp}). The general (all-$N$, all-$S$) statement remains the
nested-radical independence question.
\end{remark}

\noindent The remaining genuinely open case is a quasi-1D lattice \emph{without} any embedded ring---e.g.\
the open (Dirichlet) triangular ladder, whose spectrum is not cyclotomic; there condition~(A) holds
($d_s=1$) but (B) has no clean arithmetic and stays per-family.
